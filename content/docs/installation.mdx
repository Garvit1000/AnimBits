Before you begin, make sure you have:

- React 18.0 or later
- TypeScript 4.9 or later (recommended)
- Node.js 16.0 or later

## Quick Start

<Steps>
  <Step>
    ### Install Framer Motion
    
    AnimBits is built on top of Framer Motion. Install it in your project:
    
    <Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>

    </Tabs>
  </Step>
  
  <Step>
    ### Install Tailwind CSS (Optional)
    
    While not required, AnimBits components use Tailwind CSS for styling. If you haven't set it up:
    
    <Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>
      <Tab value="npm">
        ```bash
        npm install -D tailwindcss postcss autoprefixer
        npx tailwindcss init -p
        ```
      </Tab>
      <Tab value="pnpm">
        ```bash
        pnpm add -D tailwindcss postcss autoprefixer

        ```
      </Tab>
      <Tab value="yarn">
        ```bash
        yarn add -D tailwindcss postcss autoprefixer
        yarn dlx tailwindcss init -p
        ```
      </Tab>
      <Tab value="bun">
        ```bash
        bun add -D tailwindcss postcss autoprefixer
        bunx tailwindcss init -p
        ```
      </Tab>
    </Tabs>
    
    Configure your `tailwind.config.js`:
    
    ```js title="tailwind.config.js"
    /** @type {import('tailwindcss').Config} */
    module.exports = {
      content: [
        './pages/**/*.{js,ts,jsx,tsx,mdx}',

        './app/**/*.{js,ts,jsx,tsx,mdx}',

        extend: {},
    ```

    Add Tailwind directives to your CSS:
    
    ```css title="app/globals.css"
    @tailwind base;
    @tailwind components;
    @tailwind utilities;
    ```
  </Step>
  
  <Step>
    ### Create Directory Structure
    
    Create the necessary folders in your project:
    
    ```bash
    mkdir -p lib/hooks
    mkdir -p lib/components
    mkdir -p components/animations
    ```
    
    Your structure should look like:
    
    ```
    your-project/
    ├── lib/
    │   ├── hooks/          # Animation primitive hooks
    │   └── components/     # Generic wrapper components
    └── components/
        └── animations/     # Semantic animation components
            ├── buttons/
            ├── cards/
            ├── icons/
            └── ...
    ```
  </Step>
  
  <Step>
    ### Copy Your First Component
    
    Browse the [components](/docs/components/buttons) or [hooks](/docs/hooks/hover-lift) documentation and copy the code you need.
    
    For example, to use the hover lift animation:
    
    **Option 1: Copy the hook (recommended for flexibility)**
    
    ```tsx title="lib/hooks/use-hover-lift.ts"
    import { type MotionProps } from "framer-motion"

    export interface UseHoverLiftOptions {
      liftDistance?: number
      addShadow?: boolean
      duration?: number
      shadowStyle?: string
    }

    export function useHoverLift(options: UseHoverLiftOptions = {}): MotionProps {
      const {
        liftDistance = 8,
        addShadow = true,
        duration = 0.2,
        shadowStyle = "0 10px 30px rgba(0,0,0,0.15)",
      } = options

      return {
        initial: {
          y: 0,
          boxShadow: addShadow ? "0 0 0 rgba(0,0,0,0)" : undefined,
        },
        whileHover: {
          y: -liftDistance,
          boxShadow: addShadow ? shadowStyle : undefined,
        },
        transition: { duration },
      }
    }
    ```
    
    **Option 2: Copy the component (quick start)**
    
    ```tsx title="components/animations/buttons/lift.tsx"
    "use client"

    import { motion, type HTMLMotionProps } from "framer-motion"
    import { useHoverLift } from "@/lib/hooks/use-hover-lift"

    export interface LiftButtonProps extends Omit<HTMLMotionProps<"button">, "animate"> {
      children: React.ReactNode
      liftHeight?: number
      shadow?: boolean
    }

    export function LiftButton({
      children,
      liftHeight = 8,
      shadow = true,
      ...props
    }: LiftButtonProps) {
      const liftAnimation = useHoverLift({
        liftDistance: liftHeight,
        addShadow: shadow,
      })

      return (
        <motion.button
          {...liftAnimation}
          {...props}
        >
          {children}
        </motion.button>
      )
    }
    ```
  </Step>
  
  <Step>
    ### Use in Your Project
    
    Import and use the component or hook in your application:
    
    <Tabs items={['Using Component', 'Using Hook']}>
      <Tab value="Using Component">
        ```tsx title="app/page.tsx"
        import { LiftButton } from '@/components/animations/buttons/lift'

        export default function Home() {
          return (
            <LiftButton className="px-6 py-3 bg-blue-500 text-white rounded-lg">
              Hover me!
            </LiftButton>
          )
        }
        ```
      </Tab>
      <Tab value="Using Hook">
        ```tsx title="app/page.tsx"
        import { motion } from 'framer-motion'
        import { useHoverLift } from '@/lib/hooks/use-hover-lift'

        export default function Home() {
          const liftProps = useHoverLift({ liftDistance: 12 })
          
          return (
            <motion.div 
              {...liftProps} 
              className="p-6 bg-white rounded-lg shadow"
            >
              This card lifts on hover!
            </motion.div>
          )
        }
        ```
      </Tab>
    </Tabs>
  </Step>
</Steps>

## Framework-Specific Setup

### Next.js

AnimBits works seamlessly with Next.js 13+ (App Router). Make sure to add `"use client"` directive to components that use animations.

**Step 1: Install Framer Motion**

<Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>
  <Tab value="npm">
    ```bash
    npm install framer-motion
    ```
  </Tab>
  <Tab value="pnpm">
    ```bash
    pnpm add framer-motion
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn add framer-motion
    ```
  </Tab>
  <Tab value="bun">
    ```bash
    bun add framer-motion
    ```
  </Tab>
</Tabs>

**Step 2: Create folder structure**

```bash
mkdir -p lib/hooks lib/components components/animations
```

**Step 3: Copy hooks and components**

Copy the animation hooks you need from the documentation into `lib/hooks/`.

**Step 4: Use in your components**

```tsx title="components/animations/buttons/lift.tsx"
"use client"

import { motion } from "framer-motion"
import { useHoverLift } from "@/lib/hooks/use-hover-lift"

export function LiftButton({ children }: { children: React.ReactNode }) {
  const liftProps = useHoverLift()
  return <motion.button {...liftProps}>{children}</motion.button>
}
```

```tsx title="app/page.tsx"
import { LiftButton } from "@/components/animations/buttons/lift"

export default function Home() {
  return <LiftButton>Hover me!</LiftButton>
}
```

**Step 1: Install Framer Motion**

<Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>

  </Tab>
  <Tab value="yarn">

  </Tab>
  <Tab value="bun">
    ```bash
    bun add framer-motion
    ```
  </Tab>
</Tabs>

**Step 2: Create folder structure**

Create these folders in your project (adjust paths based on your framework):

```bash
mkdir -p lib/hooks
mkdir -p lib/components
mkdir -p components/animations
```

**Step 3: Copy animation primitives**

Browse the [hooks documentation](/docs/hooks/hover-lift) and copy the hooks you need into `lib/hooks/`.

For example, create `lib/hooks/use-hover-lift.ts`:

```tsx title="lib/hooks/use-hover-lift.ts"
import { type MotionProps } from "framer-motion"

export interface UseHoverLiftOptions {
  liftDistance?: number
  addShadow?: boolean
  duration?: number
  shadowStyle?: string
}

export function useHoverLift(options: UseHoverLiftOptions = {}): MotionProps {
  const {
    liftDistance = 8,
    addShadow = true,
    duration = 0.2,
    shadowStyle = "0 10px 30px rgba(0,0,0,0.15)",
  } = options

  return {
    initial: {
      y: 0,
      boxShadow: addShadow ? "0 0 0 rgba(0,0,0,0)" : undefined,
    },
    whileHover: {
      y: -liftDistance,
      boxShadow: addShadow ? shadowStyle : undefined,
    },
    transition: { duration },
  }
}
```

**Step 4: Use in your components**

```tsx
import { motion } from "framer-motion"
import { useHoverLift } from "./lib/hooks/use-hover-lift"

function MyComponent() {
  const liftProps = useHoverLift({ liftDistance: 12 })
  return (
    <motion.div {...liftProps} className="card">
      Content that lifts on hover
    </motion.div>
  )
}
```

**That's it!** AnimBits is framework-agnostic and works with any React setup. Just copy the code you need and customize it for your project.

## Troubleshooting

### "Module not found" errors

Make sure your path aliases are set up correctly. Check your `tsconfig.json` or bundler configuration.

### Animations not working

Ensure Framer Motion is properly installed and you're using `motion` components from `framer-motion`.

### "use client" directive missing

If using Next.js 13+ App Router, add `"use client"` at the top of any component that uses animations.

### TypeScript errors

Make sure you're using TypeScript 4.9 or later and have proper type definitions installed.
