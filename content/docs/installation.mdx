Before you begin, make sure you have:

- React 18.0 or later
- TypeScript 4.9 or later (recommended)
- Node.js 16.0 or later

    ### Copy Your First Component
    
    Browse the [components](/docs/animations/buttons/lift) or [hooks](/docs/hooks/hover-lift) documentation and copy the code you need.
    
    For example, to use the hover lift animation:
    
    **Option 1: Copy the hook (recommended for flexibility)**
    
    ```tsx title="lib/hooks/use-hover-lift.ts"
    import { type MotionProps } from "motion/react"

    export interface UseHoverLiftOptions {
      liftDistance?: number
      addShadow?: boolean
      duration?: number
      shadowStyle?: string
    }

    export function useHoverLift(options: UseHoverLiftOptions = {}): MotionProps {
      const {
        liftDistance = 8,
        addShadow = true,
        duration = 0.2,
        shadowStyle = "0 10px 30px rgba(0,0,0,0.15)",
      } = options

      return {
        initial: {
          y: 0,
          boxShadow: addShadow ? "0 0 0 rgba(0,0,0,0)" : undefined,
        },
        whileHover: {
          y: -liftDistance,
          boxShadow: addShadow ? shadowStyle : undefined,
        },
        transition: { duration },
      }
    }
    ```
    
    **Option 2: Copy the component (quick start)**
    
    ```tsx title="components/animations/buttons/lift.tsx"
    "use client"

    import { motion, type HTMLMotionProps } from "motion/react"
    import { useHoverLift } from "@/lib/use-hover-lift"

    export interface LiftButtonProps extends Omit<HTMLMotionProps<"button">, "animate"> {
      children: React.ReactNode
      liftHeight?: number
      shadow?: boolean
    }

    export function LiftButton({
      children,
      liftHeight = 8,
      shadow = true,
      ...props
    }: LiftButtonProps) {
      const liftAnimation = useHoverLift({
        liftDistance: liftHeight,
        addShadow: shadow,
      })

      return (
        <motion.button
          {...liftAnimation}
          {...props}
        >
          {children}
        </motion.button>
      )
    }
    ```
  </Step>

#### Add utility function

Create a `lib/utils.ts` file for class merging:

```ts title="lib/utils.ts"
import { type ClassValue, clsx } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}
```

#### Use in your components

Add the `"use client"` directive to any component that uses animations.

```tsx title="components/animations/buttons/lift.tsx"
"use client"

import { motion } from "motion/react"
import { useHoverLift } from "@/lib/hooks/use-hover-lift"

export function LiftButton({ children }: { children: React.ReactNode }) {
  const liftProps = useHoverLift()
  return <motion.button {...liftProps}>{children}</motion.button>
}
```

```tsx title="app/page.tsx"
import { LiftButton } from "@/components/animations/buttons/lift"

export default function Home() {
  return <LiftButton>Hover me!</LiftButton>
}
```

```ts title="vite.config.ts"
import path from 'path'

export default defineConfig({
  plugins: [react()],
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
```

```json title="tsconfig.json"
  "compilerOptions": {

}
```

```tsx title="src/App.tsx"
import { Routes, Route, Link } from 'react-router-dom'
import { motion } from "motion/react"
import { useHoverLift } from './lib/use-hover-lift'

function Home() {
  const liftProps = useHoverLift()
  
  return (
    <div className="p-8">
      <h1 className="text-3xl font-bold mb-4">Home</h1>
      <motion.button 
        {...liftProps} 
        className="px-6 py-3 bg-blue-600 text-white rounded-lg"

        Hover me!
      </motion.button>
    </div>
  )
}

function About() {
  return (
    <div className="p-8">
      <h1 className="text-3xl font-bold">About</h1>
    </div>
  )
}

function App() {
  return (
    <div>

      </nav>
      
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
      </Routes>
    </div>
  )
}

export default App
```

## Troubleshooting

### "Module not found" errors

Make sure your path aliases are set up correctly. Check your `tsconfig.json` or bundler configuration.

### Animations not working

Ensure Framer Motion is properly installed and you're using `motion` components from `framer-motion`.

### "use client" directive missing

If using Next.js 13+ App Router, add `"use client"` at the top of any component that uses animations.

### TypeScript errors

Make sure you're using TypeScript 4.9 or later and have proper type definitions installed.
