## Overview

The `useScaleIn` hook creates a scale-up entrance animation with optional fade. Elements start small and scale to full size, perfect for modals, cards, and attention-grabbing entrances.

## Installation

<Tabs items={['Manual', 'CLI']}>
  <Tab value="Manual">
    ```tsx title="lib/hooks/use-scale-in.ts"
    import { type MotionProps } from "framer-motion"

    export interface UseScaleInOptions {
      duration?: number
      delay?: number
      from?: number
      to?: number
      initialOpacity?: number
      ease?: string | number[]
      viewport?: "once" | "always"
    }

    export function useScaleIn(options: UseScaleInOptions = {}): MotionProps {
      const {
        duration = 0.5,
        delay = 0,
        from = 0.8,
        to = 1,
        initialOpacity = 0,
        ease = "easeOut",
        viewport = "once",
      } = options

      return {
        initial: {
          scale: from,
          opacity: initialOpacity,
        },
        whileInView: {
          scale: to,
          opacity: 1,
        },
        viewport: {
          once: viewport === "once",
        },
        transition: {
          duration,
          delay,
          ease,
        },
      }
    }
    ```

## Usage

### Basic Example

```tsx
import { motion } from "framer-motion"
import { useScaleIn } from "@/lib/hooks/use-scale-in"

export function Card() {
  const scaleProps = useScaleIn()
  
  return (
    <motion.div {...scaleProps} className="card">
      <h3>Scales in on view</h3>
      <p>Content appears smoothly</p>
    </motion.div>
  )
}
```

### Modal Entrance

```tsx
import { motion } from "framer-motion"
import { useScaleIn } from "@/lib/hooks/use-scale-in"

export function Modal({ isOpen }: { isOpen: boolean }) {
  const scaleProps = useScaleIn({ 
    from: 0.5,
    duration: 0.3 
  })
  
  if (!isOpen) return null
  
  return (
    <motion.div {...scaleProps} className="modal">
      <h2>Modal Title</h2>
      <p>Modal content</p>
    </motion.div>
  )
}
```

### Staggered Grid

```tsx
import { motion } from "framer-motion"
import { useScaleIn } from "@/lib/hooks/use-scale-in"

export function ProductGrid({ products }) {
  return (
    <div className="grid grid-cols-3 gap-4">
      {products.map((product, index) => {
        const scaleProps = useScaleIn({ 
          delay: index * 0.1  // Stagger effect
        })
        
        return (
          <motion.div key={product.id} {...scaleProps}>
            <img src={product.image} alt={product.name} />
            <h3>{product.name}</h3>
          </motion.div>
        )
      })}
    </div>
  )
}
```

## API Reference

### Options

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `duration` | `number` | `0.5` | Animation duration in seconds |
| `delay` | `number` | `0` | Delay before animation starts |
| `from` | `number` | `0.8` | Initial scale (0 = invisible, 1 = normal) |
| `to` | `number` | `1` | Final scale |
| `initialOpacity` | `number` | `0` | Initial opacity (0-1) |
| `ease` | `string \| number[]` | `"easeOut"` | Animation easing |
| `viewport` | `"once" \| "always"` | `"once"` | Animate once or every time in view |

## Examples

### Dramatic Entrance

```tsx
const scaleProps = useScaleIn({
  from: 0,
  duration: 0.6,
})
```

### Subtle Scale

```tsx
const scaleProps = useScaleIn({
  from: 0.95,
  duration: 0.3,
})
```

### Pop Effect

```tsx
const scaleProps = useScaleIn({
  from: 0.5,
  to: 1.05,  // Slightly overshoot
  duration: 0.4,
  ease: [0.68, -0.55, 0.265, 1.55],  // Bounce easing
})
```
