## Overview

The `useSlideIn` hook creates a slide-in entrance animation from any direction (left, right, up, down) with optional fade. Perfect for sidebars, drawers, notifications, and sequential content reveals.

<Callout type="info">
  **Directional flexibility:** Choose the slide direction that best matches your layout and user flow.
</Callout>

## Installation

<Tabs items={['Manual', 'CLI']}>
  <Tab value="Manual">
    ```tsx title="lib/hooks/use-slide-in.ts"
    import { type MotionProps } from "framer-motion"

    export type SlideDirection = "left" | "right" | "up" | "down"

    export interface UseSlideInOptions {
      direction?: SlideDirection
      distance?: number
      duration?: number
      delay?: number
      initialOpacity?: number
      ease?: string | number[]
      viewport?: "once" | "always"
    }

    export function useSlideIn(options: UseSlideInOptions = {}): MotionProps {
      const {
        direction = "left",
        distance = 50,
        duration = 0.5,
        delay = 0,
        initialOpacity = 0,
        ease = "easeOut",
        viewport = "once",
      } = options

      const getInitialPosition = () => {
        switch (direction) {
          case "left":
            return { x: -distance, y: 0 }
          case "right":
            return { x: distance, y: 0 }
          case "up":
            return { x: 0, y: -distance }
          case "down":
            return { x: 0, y: distance }
          default:
            return { x: -distance, y: 0 }
        }
      }

      const initialPosition = getInitialPosition()

      return {
        initial: {
          ...initialPosition,
          opacity: initialOpacity,
        },
        whileInView: {
          x: 0,
          y: 0,
          opacity: 1,
        },
        viewport: {
          once: viewport === "once",
        },
        transition: {
          duration,
          delay,
          ease,
        },
      }
    }
    ```

## Usage

### Basic Example

```tsx
import { motion } from "framer-motion"
import { useSlideIn } from "@/lib/hooks/use-slide-in"

export function Card() {
  const slideProps = useSlideIn({ direction: "left" })
  
  return (
    <motion.div {...slideProps} className="card">
      <h3>Slides in from left</h3>
      <p>Content appears smoothly</p>
    </motion.div>
  )
}
```

### Sidebar/Drawer

```tsx
import { motion } from "framer-motion"
import { useSlideIn } from "@/lib/hooks/use-slide-in"

export function Sidebar({ isOpen }: { isOpen: boolean }) {
  const slideProps = useSlideIn({ 
    direction: "left",
    distance: 300,
    duration: 0.3 
  })
  
  if (!isOpen) return null
  
  return (
    <motion.aside {...slideProps} className="sidebar">
      <nav>Navigation items</nav>
    </motion.aside>
  )
}
```

### Staggered List

```tsx
import { motion } from "framer-motion"
import { useSlideIn } from "@/lib/hooks/use-slide-in"

export function FeatureList({ features }) {
  return (
    <div className="space-y-4">
      {features.map((feature, index) => {
        const slideProps = useSlideIn({ 
          direction: "left",
          delay: index * 0.1 
        })
        
        return (
          <motion.div key={feature.id} {...slideProps}>
            <h3>{feature.title}</h3>
            <p>{feature.description}</p>
          </motion.div>
        )
      })}
    </div>
  )
}
```

## API Reference

### Options

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `direction` | `"left" \| "right" \| "up" \| "down"` | `"left"` | Direction to slide from |
| `distance` | `number` | `50` | Distance to slide in pixels |
| `duration` | `number` | `0.5` | Animation duration in seconds |
| `delay` | `number` | `0` | Delay before animation starts |
| `initialOpacity` | `number` | `0` | Initial opacity (0-1) |
| `ease` | `string \| number[]` | `"easeOut"` | Animation easing |
| `viewport` | `"once" \| "always"` | `"once"` | Animate once or every time in view |
