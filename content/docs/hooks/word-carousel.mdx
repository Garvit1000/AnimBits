## Overview

The `useWordCarousel` hook cycles through an array of words at a specified interval. Perfect for dynamic hero titles, feature highlights, or rotating testimonials.

<Callout type="info">
  **Works with AnimatePresence:** Pair with Framer Motion's `AnimatePresence` for smooth transitions between words.
</Callout>

## Installation

<Tabs items={['Manual', 'CLI']}>
  <Tab value="Manual">
    ```tsx title="lib/hooks/use-word-carousel.ts"
    import * as React from "react"

    export interface UseWordCarouselOptions {
      words: string[]
      interval?: number
    }

    export function useWordCarousel(options: UseWordCarouselOptions) {
      const { words, interval = 2 } = options
      const [currentIndex, setCurrentIndex] = React.useState(0)

      React.useEffect(() => {
        const timer = setInterval(() => {
          setCurrentIndex((prev) => (prev + 1) % words.length)
        }, interval * 1000)

        return () => clearInterval(timer)
      }, [words.length, interval])

      return {
        currentWord: words[currentIndex],
        currentIndex,
        key: currentIndex, // For AnimatePresence
      }
    }
    ```
  </Tab>

## Usage

### Basic Example

```tsx
import { motion, AnimatePresence } from "framer-motion"
import { useWordCarousel } from "@/lib/hooks/use-word-carousel"

export function RotatingWords() {
  const { currentWord, key } = useWordCarousel({ 
    words: ["Fast", "Modern", "Beautiful"] 
  })
  
  return (
    <div className="text-4xl font-bold">
      Build{" "}
      <AnimatePresence mode="wait">
        <motion.span
          key={key}
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          exit={{ opacity: 0, y: -20 }}
          transition={{ duration: 0.3 }}
          className="text-blue-500"
        >
          {currentWord}
        </motion.span>
      </AnimatePresence>
      {" "}Apps
    </div>
  )
}
```

### Returns

Returns an object containing:
- `currentWord` - The currently displayed word
- `currentIndex` - Index of current word in array
- `key` - Unique key for AnimatePresence (same as currentIndex)

### With Progress Indicator

```tsx
import { motion, AnimatePresence } from "framer-motion"

  })
  
  return (
    <div>

            }`}
          />
        ))}
      </div>
    </div>
  )
}
```

## Tips

### Always Use AnimatePresence

For smooth transitions, wrap the animated element in `AnimatePresence`:

```tsx
<AnimatePresence mode="wait">
  <motion.span key={key}>
    {currentWord}
  </motion.span>
</AnimatePresence>
```

### Use `mode="wait"`

This ensures the exit animation completes before the enter animation starts.

### Inline Display

For inline text, use `display: inline-block`:

```tsx
<motion.span
  key={key}
  className="inline-block"
  // ... animations
>
  {currentWord}
</motion.span>
```
