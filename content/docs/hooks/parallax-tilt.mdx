## Overview

The `useParallaxTilt` hook creates a stunning 3D tilt effect where elements rotate based on mouse position, complete with dynamic shadows for enhanced depth perception.

<Callout type="info">
  **3D Transform:** Uses CSS 3D transforms with `preserve-3d` for realistic depth effects.
</Callout>

    ```tsx title="lib/hooks/use-parallax-tilt.ts"
import * as React from "react";
import { useMotionValue, useSpring, useTransform } from "motion/react";

export interface UseParallaxTiltOptions {
  maxTilt?: number;
  duration?: number;
  shadowIntensity?: number;
  stiffness?: number;
  damping?: number;
}

export function useParallaxTilt(options: UseParallaxTiltOptions = {}) {
  const {
    maxTilt = 15,
    duration = 0.3,
    shadowIntensity = 0.3,
    stiffness = 300,
    damping = 30,
  } = options;

  const ref = React.useRef<HTMLElement>(null);
  const x = useMotionValue(0);
  const y = useMotionValue(0);

  const mouseXSpring = useSpring(x, { stiffness, damping });
  const mouseYSpring = useSpring(y, { stiffness, damping });

  const rotateX = useTransform(mouseYSpring, [-0.5, 0.5], [maxTilt, -maxTilt]);
  const rotateY = useTransform(mouseXSpring, [-0.5, 0.5], [-maxTilt, maxTilt]);

  const shadowX = useTransform(mouseXSpring, [-0.5, 0.5], [-20, 20]);
  const shadowY = useTransform(mouseYSpring, [-0.5, 0.5], [-20, 20]);

  const handleMouseMove = (e: React.MouseEvent<HTMLElement>) => {
    if (!ref.current) return;

    const rect = ref.current.getBoundingClientRect();
    const width = rect.width;
    const height = rect.height;
    const mouseX = e.clientX - rect.left;
    const mouseY = e.clientY - rect.top;
    const xPct = mouseX / width - 0.5;
    const yPct = mouseY / height - 0.5;
    x.set(xPct);
    y.set(yPct);
  };

  const handleMouseLeave = () => {
    x.set(0);
    y.set(0);
  };

  const tiltProps = {
    style: {
      rotateX,
      rotateY,
      transformStyle: "preserve-3d" as const,
      boxShadow: useTransform(
        [shadowX, shadowY],
        ([x, y]) => `${x}px ${y}px 30px rgba(0, 0, 0, ${shadowIntensity})`,
      ),
    },
    onMouseMove: handleMouseMove,
    onMouseLeave: handleMouseLeave,
    transition: { duration },
  };

  return { tiltProps, ref };
}
```

```tsx
      <div style={{ transform: "translateZ(40px)" }}>
        Floating Element
      </div>
    </motion.div>
  )
}
```

## Use cases

`useParallaxTilt` returns `{ tiltProps, ref }`. The `tiltProps` object contains a `style` with spring-driven `rotateX`, `rotateY`, `transformStyle: "preserve-3d"`, and a derived `boxShadow` MotionValue, plus `onMouseMove` and `onMouseLeave` handlers. Attach `ref` to the element so the hook can compute the relative cursor position as a fraction of element width and height.

- **Product showcase cards.** A pricing card or feature card that tilts as the cursor sweeps across it creates a sense of depth and 3D presence on an otherwise flat page.
- **Profile or team member cards.** A headshot card with `maxTilt: 10` and `shadowIntensity: 0.2` gives a subtle layered look without being distracting.
- **Portfolio project thumbnails.** Gallery images and project screenshots with a modest tilt feel like physical cards on a table.
- **Landing page hero illustrations.** Wrapping a hero image or SVG illustration in a `motion.div` with `useParallaxTilt` makes it respond to cursor movement across the entire above-the-fold area.
- **Interactive pricing tiers.** Each pricing column can tilt independently as the user moves the cursor across the page, drawing attention to the tier the cursor is nearest.

## When to use / when not to use

**Use `useParallaxTilt` directly when:**

- You need the 3D tilt on a custom element not covered by a prebuilt component.
- You want to place child elements at different `translateZ` depths using `transformStyle: "preserve-3d"` to create a true parallax depth illusion inside the tilting container.
- You need fine control over `stiffness` (default `300`) and `damping` (default `30`) to tune the spring feel.

**Avoid `useParallaxTilt` when:**

- The page is primarily a touch interface. `onMouseMove` never fires on touch, so the effect is invisible to mobile users. Disable it conditionally:

```tsx
const isTouchDevice = typeof window !== "undefined" && window.matchMedia("(hover: none)").matches
const { tiltProps, ref } = useParallaxTilt()
```

- The element contains a `<video>` or `<canvas>`. Applying `rotateX` and `rotateY` to a video can trigger a re-decode cycle in some browsers, causing a flicker.
- The element is inside a scrollable container with `overflow: hidden`. The `getBoundingClientRect` call in `onMouseMove` returns coordinates relative to the viewport, not the scroll container, which can produce incorrect tilt directions during scroll.
- The user's `prefers-reduced-motion` preference is set. 3D rotation is among the most vestibuler-discomfort-causing animation types.

## Accessibility notes

`useParallaxTilt` applies `rotateX` and `rotateY` transforms driven by pointer movement. The effect is purely cosmetic.

- The 3D rotation does not change the element's position in the tab order or its ARIA semantics.
- Respect `prefers-reduced-motion`. 3D tilting is one of the animation types most likely to trigger vestibular discomfort. Return an empty object when the preference is set:

```tsx
"use client"
import { motion } from "motion/react"
import { useReducedMotion } from "motion/react"
import { useParallaxTilt } from "@/lib/use-parallax-tilt"

export function TiltCard({ children }: { children: React.ReactNode }) {
  const reduced = useReducedMotion()
  const { tiltProps, ref } = useParallaxTilt({ maxTilt: 12, shadowIntensity: 0.25 })

  return (
    <motion.div
      ref={ref}
      {...(reduced ? {} : tiltProps)}
      className="rounded-2xl border border-neutral-200 bg-white p-6 focus-visible:outline focus-visible:outline-2 focus-visible:outline-blue-500"
    >
      {children}
    </motion.div>
  )
}
```

- On a keyboard-only workflow, the element is never in a `mousemove` interaction, so sighted keyboard users see no tilt. Ensure the card conveys its interactive state through focus styling rather than the tilt alone.
- If the tilt contains interactive children (buttons, links), confirm they remain reachable after the tilt is applied. Framer Motion applies transforms via `style` and does not alter hit-testing bounds, so click targets remain at their painted position.

## Performance considerations

- `useParallaxTilt` maintains four MotionValues (`x`, `y`, `mouseXSpring`, `mouseYSpring`) and three derived MotionValues (`rotateX`, `rotateY`, `boxShadow`). All updates flow through Framer Motion's rAF loop and do not trigger React renders.
- `rotateX` and `rotateY` are transforms, which run on the compositor. However, the derived `boxShadow` MotionValue causes a repaint on each frame when the cursor is moving. For card-sized elements this is typically one or two layers and not perceptible. For large hero elements, consider setting `shadowIntensity: 0` to disable the shadow.
- `onMouseMove` fires on every pointer event. The hook normalizes the cursor position to a fraction (`-0.5` to `0.5`) and calls `x.set()` and `y.set()`. The spring then interpolates to the new target over the next several frames. This means one `set` call per frame at most, regardless of how fast the cursor moves.
- The `transformStyle: "preserve-3d"` style creates a stacking context. Descendants with absolute positioning that extend outside the element's bounds may be clipped by the 3D rendering layer. Test child overflow before deploying.

With `transformStyle: "preserve-3d"` already set by the hook, child elements can use `translateZ` to float at different depths:

```tsx

import { useParallaxTilt } from "@/lib/use-parallax-tilt"

export function DepthCard() {
  const { tiltProps, ref } = useParallaxTilt({ maxTilt: 15 })

  return (
    <motion.div ref={ref} {...tiltProps} className="relative h-64 w-64 rounded-2xl bg-gradient-to-br from-indigo-500 to-purple-600">
      <div style={{ transform: "translateZ(0px)" }} className="absolute inset-0 rounded-2xl p-6">
        <p className="text-white/60 text-sm">Background layer</p>
      </div>
      <div style={{ transform: "translateZ(30px)" }} className="absolute bottom-6 left-6">
        <p className="font-bold text-white text-lg">Front layer</p>
      </div>

      </div>
    </motion.div>
  )
}
```

## Real-world examples

### Pricing tier card with tilt

```tsx
"use client"
import { motion } from "motion/react"
import { useReducedMotion } from "motion/react"
import { useParallaxTilt } from "@/lib/use-parallax-tilt"

interface Tier {
  name: string
  price: string
  features: string[]
  highlighted?: boolean
}

export function PricingCard({ tier }: { tier: Tier }) {
  const reduced = useReducedMotion()
  const { tiltProps, ref } = useParallaxTilt({
    maxTilt: 10,
    shadowIntensity: tier.highlighted ? 0.35 : 0.2,
    stiffness: 280,
    damping: 28,
  })

  return (
    <motion.div
      ref={ref}
      {...(reduced ? {} : tiltProps)}
      className={`rounded-2xl border p-8 ${
        tier.highlighted
          ? "border-blue-500 bg-blue-600 text-white"
          : "border-neutral-200 bg-white dark:border-neutral-800 dark:bg-neutral-900"
      }`}
    >
      <h3 className="text-lg font-bold">{tier.name}</h3>
      <p className="mt-2 text-4xl font-black">{tier.price}</p>
      <ul className="mt-6 space-y-2">
        {tier.features.map((f) => (
          <li key={f} className="flex items-center gap-2 text-sm">
            <span className={tier.highlighted ? "text-blue-200" : "text-green-500"}>+</span>
            {f}
          </li>
        ))}
      </ul>
    </motion.div>
  )
}
```
