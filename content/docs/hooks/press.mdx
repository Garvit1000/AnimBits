## Overview

The `usePress` hook adds a press-down effect when clicking or tapping elements, providing tactile feedback. Perfect for buttons, cards, and any interactive element that needs visual confirmation of user interaction.

<Callout type="info">
  **Tactile Feedback:** This hook uses `whileTap` to provide immediate visual feedback, making your UI feel more responsive and interactive.
</Callout>

import { type MotionProps, type Easing } from "framer-motion";

export interface UsePressOptions {
  pressScale?: number;

  duration?: number;

  ease?: Easing | Easing[];

  addRotation?: boolean;

  rotationAmount?: number;
}

export function usePress(options: UsePressOptions = {}): MotionProps {
  const {
    pressScale = 0.95,
    duration = 0.1,
    ease = "easeInOut",
    addRotation = false,
    rotationAmount = 2,
  } = options;

  return {
    whileTap: {
      scale: pressScale,
      rotate: addRotation ? rotationAmount : 0,
    },
    transition: {
      duration,
      ease,
    },
  };
}

## Usage

### Basic Example

```tsx
import { motion } from "framer-motion"
import { usePress } from "@/lib/hooks/use-press"

export function PressButton() {
  const pressProps = usePress()
  
  return (
    <motion.button 
      {...pressProps}
      className="inline-flex h-11 items-center gap-2 rounded-lg bg-blue-600 px-6 text-sm font-semibold text-white"
    >
      Click me
    </motion.button>
  )
}
```

### With Custom Scale

```tsx
import { motion } from "framer-motion"
import { usePress } from "@/lib/hooks/use-press"

export function StrongPress() {
  const pressProps = usePress({ 
    pressScale: 0.9,  // More dramatic press
    duration: 0.15 
  })
  
  return (
    <motion.button {...pressProps} className="btn">
      Strong press effect
    </motion.button>
  )
}
```

### With Rotation

```tsx
import { motion } from "framer-motion"
import { usePress } from "@/lib/hooks/use-press"

export function RotatingPress() {
  const pressProps = usePress({ 
    addRotation: true,
    rotationAmount: 3 
  })
  
  return (
    <motion.button {...pressProps} className="btn">
      Press with rotation
    </motion.button>
  )
}
```

### Returns

Returns `MotionProps` object containing:
- `whileTap` - Animation state while element is pressed
- `transition` - Animation transition config

## Composing with Other Hooks

Combine with hover effects for rich interactions:

```tsx
import { usePress } from "@/lib/hooks/use-press"
import { useHoverLift } from "@/lib/hooks/use-hover-lift"

export function RichButton() {
  const press = usePress({ pressScale: 0.92 })
  const lift = useHoverLift({ liftDistance: 6 })
  
  // Merge the animations
  const combined = {
    ...lift,
    ...press,
  }
  
  return (
    <motion.button {...combined} className="btn">
      Hover + Press
    </motion.button>
  )
}
```

## Accessibility

- Press feedback is visual only and doesn't affect keyboard navigation
- Ensure buttons remain accessible via keyboard (Enter/Space keys)
- Consider adding focus states for keyboard users
- The animation is fast enough not to interfere with rapid clicking

## Browser Support

Works in all modern browsers that support CSS transforms and Framer Motion:
- Chrome/Edge 90+
- Firefox 88+
- Safari 14+
- Mobile browsers (iOS Safari, Chrome Mobile)

<Callout type="warn">
  **Touch devices:** The `whileTap` prop works on both mouse clicks and touch events, making it perfect for mobile-first designs.
</Callout>
