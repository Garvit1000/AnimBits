## Overview

The `useScramble` hook creates a "decoding" text animation where random characters scramble before revealing the final text. Perfect for tech-themed interfaces, loading states, or dramatic text reveals.

<Callout type="info">
  **Pure React:** No motion library required - uses React state and intervals for the animation.
</Callout>

## Installation

<Tabs items={['Manual', 'CLI']}>
  <Tab value="Manual">
    ```tsx title="lib/hooks/use-scramble.ts"
    import * as React from "react"    export interface UseScrambleOptions {      text?: string      duration?: number      delay?: number      scrambleChars?: string    }    export function useScramble(options: UseScrambleOptions): string {      const {        text = "",        duration = 0.8,        delay = 0,        scrambleChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()",      } = options      const [displayText, setDisplayText] = React.useState(text)      React.useEffect(() => {        const iterations = Math.floor(duration * 60)         let currentIteration = 0        const timeout = setTimeout(() => {          const interval = setInterval(() => {            if (currentIteration >= iterations) {              setDisplayText(text)              clearInterval(interval)              return            }            const progress = currentIteration / iterations            const newText = text              .split("")              .map((char, index) => {                if (char === " ") return " "                if (index / text.length < progress) {                  return char                }                return scrambleChars[Math.floor(Math.random() * scrambleChars.length)]              })              .join("")            setDisplayText(newText)            currentIteration++          }, (duration * 1000) / iterations)          return () => clearInterval(interval)        }, delay * 1000)        return () => clearTimeout(timeout)      }, [text, duration, delay, scrambleChars])      return displayText    }    ```

## Usage

### Basic Example

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function ScrambleHeading() {  const displayText = useScramble({ text: "Hello World" })  return (    <h1 className="font-mono text-4xl">      {displayText}    </h1>  )}```

### With Custom Duration

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function SlowScramble() {  const displayText = useScramble({     text: "Decoding...",    duration: 2,  })  return (    <div className="font-mono text-green-500">      {displayText}    </div>  )}```

### With Delay

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function DelayedScramble() {  const displayText = useScramble({     text: "Welcome Back",    duration: 1,    delay: 0.5,   })  return <h2 className="font-mono">{displayText}</h2>}```

## Examples

### Hero Title

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function Hero() {  const title = useScramble({     text: "Welcome to the Future",    duration: 1.5,  })  return (    <section className="hero">      <h1 className="font-mono text-6xl font-bold">        {title}      </h1>    </section>  )}```

### Loading State

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function LoadingScramble() {  const [isLoading, setIsLoading] = React.useState(true)  const text = useScramble({     text: isLoading ? "LOADING..." : "COMPLETE",    duration: 0.6,  })  return (    <div className="font-mono text-green-400">      {text}    </div>  )}```

### Staggered Multiple Lines

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function StaggeredScramble() {  const line1 = useScramble({ text: "System Initialized", delay: 0 })  const line2 = useScramble({ text: "Loading Modules", delay: 0.8 })  const line3 = useScramble({ text: "Ready", delay: 1.6 })  return (    <div className="font-mono space-y-2">      <div>{line1}</div>      <div>{line2}</div>      <div>{line3}</div>    </div>  )}```

### Matrix Effect

```tsx
import { useScramble } from "@/lib/hooks/use-scramble"export function MatrixText() {  const text = useScramble({     text: "ENTER THE MATRIX",    duration: 2,    scrambleChars: "ｱｲｳｴｵｶｷｸｹｺｻｼｽｾｿﾀﾁﾂﾃﾄﾅﾆﾇﾈﾉﾊﾋﾌﾍﾎﾏﾐﾑﾒﾓﾔﾕﾖﾗﾘﾙﾚﾛﾜﾝ01",  })  return (    <h1 className="font-mono text-green-500 text-4xl">      {text}    </h1>  )}```

## Styling Tips

Works best with monospace fonts:

```tsx
<div className="font-mono">  {displayText}</div>```

Add a cursor effect:

```tsx
<div className="font-mono">  {displayText}<span className="animate-pulse">_</span></div>```

## Performance

- Lightweight - no animation library required
- Uses `setInterval` for smooth 60fps animation
- Automatically cleans up on unmount

## Use Cases

- Tech/cyberpunk themed interfaces
- Loading states
- Dramatic text reveals
- Terminal/console effects
- Hacker/coding aesthetics
- Data decryption visualizations
