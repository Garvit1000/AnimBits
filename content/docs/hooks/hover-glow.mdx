## Overview

The `useHoverGlow` hook adds a colored glow effect (via box-shadow) when hovering over an element. Perfect for creating attention-grabbing interactive elements.

## Installation

<Tabs items={['Manual', 'CLI']}>
  <Tab value="Manual">
    Copy the hook into your project:

    ```tsx title="lib/hooks/use-hover-glow.ts"
import { type MotionProps, type Easing } from "framer-motion";

export interface UseHoverGlowOptions {
  glowColor?: string;

  glowBlur?: number;

  glowSpread?: number;

  duration?: number;

  ease?: Easing | Easing[];

  scale?: number;

  innerGlow?: boolean;
}

export function useHoverGlow(options: UseHoverGlowOptions = {}): MotionProps {
  const {
    glowColor = "rgba(59, 130, 246, 0.5)",
    glowBlur = 20,
    glowSpread = 5,
    duration = 0.3,
    ease = "easeOut",
    scale = 1.02,
    innerGlow = false,
  } = options;

  const baseShadow = "0 2px 8px rgba(0, 0, 0, 0.1)";
  const glowShadow = `0 0 ${glowBlur}px ${glowSpread}px ${glowColor}`;
  const innerGlowShadow = innerGlow
    ? `inset 0 0 ${glowBlur / 2}px ${glowColor}`
    : "";

  const hoverShadow = innerGlow
    ? `${baseShadow}, ${glowShadow}, ${innerGlowShadow}`
    : `${baseShadow}, ${glowShadow}`;

  return {
    initial: {
      scale: 1,
      boxShadow: baseShadow,
    },
    whileHover: {
      scale,
      boxShadow: hoverShadow,
    },
    transition: {
      duration,
      ease,
    },
  };
}
```

## Usage

### Basic Example

```tsx
import { motion } from "framer-motion"
import { useHoverGlow } from "@/lib/hooks/use-hover-glow"

export function GlowButton() {
  const glowProps = useHoverGlow()
  
  return (
    <motion.button 
      {...glowProps}
      className="inline-flex h-11 items-center gap-2 rounded-lg bg-blue-600 px-6 text-sm font-semibold text-white"
    >
      Hover for glow
    </motion.button>
  )
}
```

### Custom Colors

```tsx
import { motion } from "framer-motion"
import { useHoverGlow } from "@/lib/hooks/use-hover-glow"

export function ColoredGlows() {
  const blueGlow = useHoverGlow({ 
    glowColor: "rgba(59, 130, 246, 0.6)" 
  })
  
  const purpleGlow = useHoverGlow({ 
    glowColor: "rgba(168, 85, 247, 0.6)" 
  })
  
  const greenGlow = useHoverGlow({ 
    glowColor: "rgba(34, 197, 94, 0.6)" 
  })
  
  return (
    <div className="flex gap-4">
      <motion.button {...blueGlow}>Blue</motion.button>
      <motion.button {...purpleGlow}>Purple</motion.button>
      <motion.button {...greenGlow}>Green</motion.button>
    </div>
  )
}
```

## API Reference

### Options

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `glowColor` | `string` | `"rgba(59, 130, 246, 0.5)"` | CSS color for the glow |
| `glowSize` | `number` | `20` | Glow spread in pixels |
| `duration` | `number` | `0.3` | Animation duration in seconds |
| `ease` | `string \| number[]` | `"easeOut"` | Animation easing |

## Examples

### Intense Glow

```tsx
const glowProps = useHoverGlow({
  glowColor: "rgba(239, 68, 68, 0.8)",
  glowSize: 40,
})
```

### Subtle Glow

```tsx
const glowProps = useHoverGlow({
  glowColor: "rgba(59, 130, 246, 0.3)",
  glowSize: 10,
  duration: 0.2,
})
```

### Combine with Lift

```tsx
import { useHoverLift } from "@/lib/hooks/use-hover-lift"
import { useHoverGlow } from "@/lib/hooks/use-hover-glow"

const lift = useHoverLift()
const glow = useHoverGlow({ glowColor: "rgba(168, 85, 247, 0.5)" })

const combined = {
  ...lift,
  whileHover: {
    ...lift.whileHover,
    ...glow.whileHover,
  },
}

<motion.div {...combined}>Lift + Glow!</motion.div>
```
