## Overview

The `ExpandingCard` component creates a seamless transition from a compact grid view to an immersive detail view. Choose from multiple animation variants to control how cards expand.

    ```tsx title="components/animations/transitions/expanding-card.tsx"
    "use client";

    import React, { useState, useId } from "react";
    import { motion, AnimatePresence } from "motion/react";
    import { X } from "lucide-react";
    import { cn } from "@/lib/utils";

    export interface CardItem {
      id: number | string;
      title: string;
      description: string;
      longDescription: string;
      image: string;
    }

    interface ExpandingCardProps {
      items: CardItem[];
      className?: string;
      columns?: 2 | 3 | 4;
      variant?: "scale-center" | "slide-up" | "rotate-scale" | "elastic" | "fade-scale";
    }

    export function ExpandingCard({ items, className, columns = 3, variant = "scale-center" }: ExpandingCardProps) {
      const [selectedId, setSelectedId] = useState<number | string | null>(null);
      const id = useId();

      const gridCols = {
        2: "md:grid-cols-2",
        3: "md:grid-cols-3",
        4: "md:grid-cols-4"
      }[columns];

      const getExpandTransition = () => {
        switch (variant) {
          case "scale-center":
            return { type: "spring" as const, bounce: 0.2, duration: 0.6 };
          case "slide-up":
            return { type: "spring" as const, bounce: 0.15, duration: 0.5 };
          case "rotate-scale":
            return { type: "spring" as const, bounce: 0.25, duration: 0.7 };
          case "elastic":
            return { type: "spring" as const, bounce: 0.4, duration: 0.8 };
          case "fade-scale":
            return { type: "tween" as const, duration: 0.3, ease: "easeOut" as const };
          default:
            return { type: "spring" as const, bounce: 0.2, duration: 0.6 };
        }
      };

      const getExpandInitial = () => {
        switch (variant) {
          case "slide-up":
            return { opacity: 0, y: 50 };
          case "rotate-scale":
            return { opacity: 0, scale: 0.8, rotate: -5 };
          case "fade-scale":
            return { opacity: 0, scale: 0.95 };
          default:
            return {};
        }
      };

      const getExpandAnimate = () => {
        switch (variant) {
          case "slide-up":
            return { opacity: 1, y: 0 };
          case "rotate-scale":
            return { opacity: 1, scale: 1, rotate: 0 };
          case "fade-scale":
            return { opacity: 1, scale: 1 };
          default:
            return {};
        }
      };

      return (
        <div className={cn("flex flex-col items-center justify-center min-h-[600px] w-full p-8 bg-zinc-50 dark:bg-zinc-900 rounded-xl overflow-hidden relative", className)}>
          <div className={cn("grid grid-cols-1 gap-4 w-full max-w-6xl relative z-10", gridCols)}>
            {items.map((card) => (
              <motion.div
                key={card.id}
                layoutId={`${id}-card-${card.id}`}
                onClick={() => setSelectedId(card.id)}
                className={cn(
                  "relative overflow-hidden rounded-xl cursor-pointer shadow-lg hover:shadow-xl transition-shadow bg-white dark:bg-zinc-800",
                  "flex flex-col h-64 group"
                )}
                whileHover={{ scale: 1.02 }}
                whileTap={{ scale: 0.98 }}
              >
                <motion.div
                  layoutId={`${id}-image-${card.id}`}
                  className="h-40 w-full overflow-hidden"
                >
                  <img
                    src={card.image}
                    alt={card.title}
                    className="w-full h-full object-cover transition-transform duration-500 group-hover:scale-110"
                  />
                </motion.div>
                <div className="p-4">
                  <motion.h3
                    layoutId={`${id}-title-${card.id}`}
                    className="text-lg font-bold text-zinc-900 dark:text-zinc-100"
                  >
                    {card.title}
                  </motion.h3>
                  <motion.p
                    layoutId={`${id}-desc-${card.id}`}
                    className="text-sm text-zinc-500 dark:text-zinc-400 mt-1"
                  >
                    {card.description}
                  </motion.p>
                </div>
              </motion.div>
            ))}
          </div>

          <AnimatePresence>
            {selectedId && (
              <>
                <motion.div
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                  exit={{ opacity: 0 }}
                  onClick={() => setSelectedId(null)}
                  className="absolute inset-0 bg-black/50 backdrop-blur-sm z-20"
                />
                <div className="absolute inset-0 flex items-center justify-center z-30 pointer-events-none p-4">
                  {items
                    .filter((c) => c.id === selectedId)
                    .map((card) => (
                      <motion.div
                        key={card.id}
                        layoutId={`${id}-card-${card.id}`}
                        className="w-full max-w-lg bg-white dark:bg-zinc-900 rounded-2xl overflow-hidden shadow-2xl pointer-events-auto relative max-h-[85vh] flex flex-col"
                        transition={getExpandTransition()}
                        initial={getExpandInitial()}
                        animate={getExpandAnimate()}
                      >
                        <motion.button
                          className="absolute top-4 right-4 p-2 bg-black/10 dark:bg-white/10 rounded-full text-zinc-900 dark:text-zinc-100 z-50 hover:bg-black/20 dark:hover:bg-white/20 transition-colors"
                          onClick={(e) => {
                            e.stopPropagation();
                            setSelectedId(null);
                          }}
                          initial={{ opacity: 0 }}
                          animate={{ opacity: 1, transition: { delay: 0.2 } }}
                          exit={{ opacity: 0 }}
                        >
                          <X size={20} />
                        </motion.button>

                        <motion.div
                          layoutId={`${id}-image-${card.id}`}
                          className="h-64 w-full overflow-hidden flex-shrink-0"
                        >
                          <img
                            src={card.image}
                            alt={card.title}
                            className="w-full h-full object-cover"
                          />
                        </motion.div>

                        <div className="p-6 overflow-y-auto">
                          <motion.h3
                            layoutId={`${id}-title-${card.id}`}
                            className="text-2xl font-bold text-zinc-900 dark:text-zinc-100 mb-2"
                          >
                            {card.title}
                          </motion.h3>
                          <motion.p
                            layoutId={`${id}-desc-${card.id}`}
                            className="text-base text-zinc-600 dark:text-zinc-400 mb-4"
                          >
                            {card.description}
                          </motion.p>

                          <motion.div
                            initial={{ opacity: 0, y: 20 }}
                            animate={{ opacity: 1, y: 0 }}
                            exit={{ opacity: 0, y: 10 }}
                            transition={{ delay: 0.1 }}
                            className="text-sm text-zinc-500 dark:text-zinc-400 leading-relaxed"
                          >
                            <p>{card.longDescription}</p>
                          </motion.div>
                        </div>
                      </motion.div>
                    ))}
                </div>
              </>
            )}
          </AnimatePresence>
        </div>
      );
    }
    ```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `CardItem[]` | - | Array of card data objects. |
| `columns` | `2 \| 3 \| 4` | `3` | Number of columns in grid layout. |
| `variant` | `"scale-center" \| "slide-up" \| "rotate-scale" \| "elastic" \| "fade-scale"` | `"scale-center"` | Animation style for card expansion. |
| `className` | `string` | - | Additional CSS classes. |

## Use cases

`ExpandingCard` is built around Framer Motion's shared-layout system: the grid card and the expanded detail view share four `layoutId` values each (`card-{id}`, `image-{id}`, `title-{id}`, `desc-{id}`), so the browser morphs between them rather than cutting or fading. Use it when visual continuity between a thumbnail and its full-size detail view matters:

- **Photo and portfolio galleries.** A grid of cropped thumbnails that each expand into a hero image with a caption. The shared image `layoutId` keeps the photo in place while the card grows.
- **E-commerce product grids.** A compact product card expands to show a longer description, size options, or specs without navigating away.
- **Pricing or FAQ accordions.** A collapsed feature card (title + one-liner) expands into a full benefit explanation. `variant="fade-scale"` suits text-heavy content where the tween easing (0.3s easeOut) avoids spring overshoot on text.
- **Portfolio case studies.** A thumbnail with a project title expands into a full case-study panel. `variant="elastic"` (bounce 0.4) adds energy appropriate for creative work; drop it to `variant="scale-center"` (bounce 0.2) for professional or finance contexts.
- **Content cards in a CMS dashboard.** Editors click a post card in a grid to expand it and edit metadata inline.

## When to use / when not to use

**Use `ExpandingCard` when:**

- The user needs visual continuity between a compact representation and its detail view. The shared `layoutId` morph communicates "you are looking at the same item, expanded", which a simple modal fade cannot.
- Your data is already structured as a flat list of `CardItem` objects with a `longDescription` field distinct from `description`. The component renders both in the detail view.
- You have 3 to 12 cards. Fewer than three makes the grid feel empty; many more means users will be scrolling past unmounted `AnimatePresence` trees.

**Do not use `ExpandingCard` when:**

- You need deep interactivity inside the expanded view (forms with multiple steps, nested navigations). The expanded panel sits inside a `motion.div` with no focus trap or scroll lock, so keyboard and screen-reader users cannot safely interact with it.
- The primary action is to navigate to a separate page. Reach for a standard `<Link>` card instead; shared-layout morphing feels misleading when the URL changes without continuity.
- Cards contain wildly different aspect-ratio images. The shared `layoutId` on the image container forces it to scale, which can distort tall portrait images during the tween.
- You want a plain dialog without the layout-morph effect. The [Shared Modal](/docs/transitions/shared-modal) component gives you the same `AnimatePresence` wrapper without coupling the trigger and dialog via shared `layoutId`.

## Accessibility notes

The current implementation uses plain `motion.div` elements for both the grid card and the overlay. There are several gaps to address before deploying in a production app:

- **Dialog semantics.** The expanded overlay is visually a dialog but carries no `role="dialog"` or `aria-modal="true"`. Add both to the expanded `motion.div` so screen readers announce it as a dialog and suppress background content.
- **Focus trap.** When the overlay opens, focus should move into the expanded card and be trapped there. Libraries like `focus-trap-react` work alongside Framer Motion without conflict.
- **Escape to close.** Handle `keydown` for `Escape` to call `setSelectedId(null)`. Without it, keyboard users cannot dismiss the overlay.
- **Return focus.** When the overlay closes, return focus to the card that triggered the expansion. Store a `ref` to the trigger before opening and call `.focus()` inside the `onAnimationComplete` callback on exit.
- **Close button label.** The `<X size={20} />` button has no visible text. Add `aria-label="Close"` to it.
- **Backdrop click.** The backdrop `motion.div` calls `setSelectedId(null)` on click, but it should also be keyboard-reachable (`role="button"` or a hidden button element).
- **Scroll lock.** Add `overflow: hidden` to `document.body` while the overlay is open so background content does not scroll behind the backdrop.

## Performance considerations

- **Shared layout measurement.** Framer Motion measures and animates layout for every element sharing a `layoutId`. Each card holds four shared elements (`card`, `image`, `title`, `desc`). With twelve cards in the grid, that is 48 layout measurements on each expand or collapse. Keep the grid to a reasonable size and avoid nesting many other `layout`-animated elements inside the same render tree.
- **`AnimatePresence` and mount cost.** The overlay is conditionally rendered inside `AnimatePresence`. It is fully mounted on open and unmounted on close, so you pay a render cost each time. This is fine for the use case but means heavy content (e.g., `<video>` elements) inside `longDescription` will remount and restart on every open.
- **Spring settling.** The `elastic` variant uses `bounce: 0.4` with a `duration: 0.8s` spring. Springs do not have a strict end time; with high bounce they can settle over 1 to 1.5 seconds. Use `bounce: 0.2` (the `scale-center` default) if users complain the animation feels slow.
- **Backdrop `backdrop-blur-sm`.** CSS `backdrop-filter` forces the GPU to composite the blurred layer separately. On mobile devices with many layers, this can cause jank. Test on low-end targets and remove the blur if frame rate drops.
- **`useId()` prefix.** The component calls `useId()` to namespace all `layoutId` values, preventing collisions if you mount multiple `ExpandingCard` instances on the same page. Do not replace these with static strings.
