## Overview

The `ZoomHollowText` component creates a stunning text transition with zoom-in animation and hollow outline effect. Perfect for hero sections, theme toggles, and attention-grabbing headlines. Works seamlessly with light and dark modes.

    ```tsx title="components/animations/transitions/zoom-hollow.tsx"
    "use client";

    import React, { useState, useEffect } from "react";
    import { motion, AnimatePresence } from "motion/react";
    import { cn } from "@/lib/utils";

    interface ZoomHollowTextProps {
      words: string[];
      className?: string;
      duration?: number;
      interval?: number;
      strokeWidth?: number;
      fillColor?: string;
      strokeColor?: string;
    }

    export function ZoomHollowText({
      words,
      className = "",
      duration = 0.8,
      interval = 2500,
      strokeWidth = 2,
      fillColor = "transparent",
      strokeColor = "currentColor",
    }: ZoomHollowTextProps) {
      const [index, setIndex] = useState(0);

      useEffect(() => {
        const timer = setInterval(() => {
          setIndex((prev) => (prev + 1) % words.length);
        }, interval);
        return () => clearInterval(timer);
      }, [words.length, interval]);

      return (
        <div className={cn("relative inline-block", className)}>
          <AnimatePresence mode="wait">
            <motion.span
              key={words[index]}
              initial={{ scale: 0.5, opacity: 0, filter: "blur(10px)" }}
              animate={{ scale: 1, opacity: 1, filter: "blur(0px)" }}
              exit={{ scale: 1.5, opacity: 0, filter: "blur(10px)" }}
              transition={{
                duration,
                ease: [0.22, 1, 0.36, 1],
              }}
              style={{
                WebkitTextStroke: `${strokeWidth}px ${strokeColor}`,
                WebkitTextFillColor: fillColor,
                paintOrder: "stroke fill",
              }}
              className="block"
            >
              {words[index]}
            </motion.span>
          </AnimatePresence>
        </div>
      );
    }
    ```

## Usage

```tsx
import { ZoomHollowText } from "@/components/animbits/zoom-hollow";

export default function Example() {
  return (
    <ZoomHollowText
      words={["Design", "Create", "Innovate", "Build"]}
      className="text-7xl font-black"
    />
  );
}
```
## Examples

### Custom Colors

```tsx
<ZoomHollowText
  words={["Bold", "Bright", "Beautiful"]}
  className="text-6xl font-black"
  strokeColor="#3b82f6"
  fillColor="transparent"
  strokeWidth={3}
/>
```

### Faster Animation

```tsx
<ZoomHollowText
  words={["Fast", "Quick", "Rapid"]}
  className="text-5xl font-bold"
  duration={0.5}
  interval={1500}
/>
```

### Filled Text

```tsx
<ZoomHollowText
  words={["Solid", "Bold", "Strong"]}
  className="text-6xl font-black"
  fillColor="currentColor"
  strokeWidth={0}
/>
```

### Theme-Aware Colors

The component automatically adapts to your theme using `currentColor`:

```tsx
<ZoomHollowText
  words={["Light", "Dark", "Auto"]}
  className="text-6xl font-black text-foreground"
  strokeColor="currentColor"
/>
```

## CSS Customization

The hollow effect is created using webkit text stroke. You can customize it with CSS:

```css
.custom-hollow {
  -webkit-text-stroke: 3px currentColor;
  -webkit-text-fill-color: transparent;
  paint-order: stroke fill;
}
```

For dark mode support, use CSS variables:

```css
.theme-aware-hollow {
  -webkit-text-stroke: 2px hsl(var(--foreground));
  -webkit-text-fill-color: transparent;
}

```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `words` | `string[]` | Required | Array of words to cycle through |
| `className` | `string` | `""` | Additional CSS classes |
| `duration` | `number` | `0.8` | Animation duration in seconds |
| `interval` | `number` | `2500` | Time between word changes in ms |
| `strokeWidth` | `number` | `2` | Width of the text outline |
| `fillColor` | `string` | `"transparent"` | Fill color of the text |
| `strokeColor` | `string` | `"currentColor"` | Color of the text outline |
