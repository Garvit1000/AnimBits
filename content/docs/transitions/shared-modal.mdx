## Overview

The `SharedModal` uses shared layout animations to morph a trigger element into a modal dialog. Choose from multiple animation variants to match your design aesthetic.

    ```tsx title="components/animations/transitions/shared-modal.tsx"
    "use client";

    import React, { useState, useId } from "react";
    import { motion, AnimatePresence } from "motion/react";
    import { cn } from "@/lib/utils";
    import { Plus } from "lucide-react";

    interface SharedModalProps {
      variant?: "scale" | "slide-bottom" | "slide-right" | "fade-in" | "scale-up";
    }

    export function SharedModal({ variant = "scale" }: SharedModalProps = {}) {
      const [isOpen, setIsOpen] = useState(false);
      const id = useId();

      const getModalTransition = () => {
        switch (variant) {
          case "scale":
            return { type: "spring" as const, bounce: 0.2, duration: 0.6 };
          case "slide-bottom":
            return { type: "spring" as const, bounce: 0.15, duration: 0.5 };
          case "slide-right":
            return { type: "spring" as const, bounce: 0.15, duration: 0.5 };
          case "fade-in":
            return { type: "tween" as const, duration: 0.3, ease: "easeOut" as const };
          case "scale-up":
            return { type: "spring" as const, bounce: 0.5, duration: 0.8 };
          default:
            return { type: "spring" as const, bounce: 0.2, duration: 0.6 };
        }
      };

      const getModalInitial = () => {
        switch (variant) {
          case "slide-bottom":
            return { opacity: 0, y: "100%" };
          case "slide-right":
            return { opacity: 0, x: "100%" };
          case "fade-in":
            return { opacity: 0, scale: 1.1 };
          case "scale-up":
            return { opacity: 0, scale: 0.5 };
          default:
            return {};
        }
      };

      const getModalAnimate = () => {
        switch (variant) {
          case "slide-bottom":
            return { opacity: 1, y: 0 };
          case "slide-right":
            return { opacity: 1, x: 0 };
          case "fade-in":
            return { opacity: 1, scale: 1 };
          case "scale-up":
            return { opacity: 1, scale: 1 };
          default:
            return {};
        }
      };

      return (
        <div className="flex items-center justify-center p-8 bg-zinc-50 dark:bg-zinc-900 rounded-xl min-h-[400px] w-full">
          <AnimatePresence>
            {!isOpen ? (
              <motion.div
                layoutId={`${id}-modal`}
                onClick={() => setIsOpen(true)}
                className="h-16 w-16 bg-white dark:bg-zinc-800 rounded-full shadow-lg cursor-pointer flex items-center justify-center border border-zinc-200 dark:border-zinc-700"
                whileHover={{ scale: 1.05 }}
                whileTap={{ scale: 0.95 }}
              >
                <motion.div layoutId={`${id}-icon`}>
                  <Plus className="text-zinc-500 dark:text-zinc-400" />
                </motion.div>
              </motion.div>
            ) : (
              <div className={cn(
                "fixed inset-0 z-[100] px-4 pointer-events-none",
                (variant === "slide-bottom" || variant === "slide-right") ? "flex" : "grid place-items-center",
                variant === "slide-bottom" && "items-end justify-center pb-4",
                variant === "slide-right" && "items-center justify-end pr-4"
              )}>
                <motion.div
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                  exit={{ opacity: 0 }}
                  onClick={() => setIsOpen(false)}
                  className="absolute inset-0 bg-black/20 backdrop-blur-sm pointer-events-auto"
                />
                <motion.div
                  {...(variant === "scale" ? { layoutId: `${id}-modal` } : {})}
                  className="w-full max-w-sm bg-white dark:bg-zinc-800 rounded-2xl shadow-2xl overflow-hidden z-20 relative pointer-events-auto"
                  transition={getModalTransition()}
                  initial={getModalInitial()}
                  animate={getModalAnimate()}
                  exit={getModalInitial()}
                  style={{ transformPerspective: 1000 }}
                >
                  <motion.div layoutId={`${id}-icon`} className="absolute top-4 right-4 z-10">
                    <button
                      onClick={(e) => { e.stopPropagation(); setIsOpen(false); }}
                      className="p-1 rounded-full bg-black/10 dark:bg-white/10 hover:bg-black/20 dark:hover:bg-white/20 transition-colors"
                    >
                      <Plus className="rotate-45 text-zinc-500 dark:text-zinc-400" />
                    </button>
                  </motion.div>

                  <img 
                    src="https://picsum.photos/seed/modal1/400/200" 
                    alt="Project header"
                    className="h-32 w-full object-cover"
                  />
                  <div className="p-6">
                    <h3 className="text-xl font-bold text-zinc-900 dark:text-zinc-100 mb-2">New Project</h3>
                    <p className="text-zinc-500 dark:text-zinc-400 mb-6">Create a new project to get started with your ideas.</p>
                    <div className="flex justify-end gap-2">
                      <button
                        onClick={(e) => { e.stopPropagation(); setIsOpen(false); }}
                        className="px-4 py-2 text-sm font-medium text-zinc-600 dark:text-zinc-300 hover:bg-zinc-100 dark:hover:bg-zinc-700 rounded-lg transition-colors"
                      >
                        Cancel
                      </button>

                        onClick={(e) => { e.stopPropagation(); setIsOpen(false); }}
                        className="px-4 py-2 text-sm font-medium bg-zinc-900 dark:bg-zinc-100 text-white dark:text-zinc-900 rounded-lg hover:opacity-90 transition-opacity"
                      >
                        Create
                      </button>
                    </div>
    ```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"scale" \| "slide-bottom" \| "slide-right" \| "fade-in" \| "scale-up"` | `"scale"` | Animation style for modal opening. |

## Use cases

`SharedModal` wraps both the trigger element and the modal in a single `AnimatePresence`, toggling between them based on the `isOpen` boolean. In the `"scale"` variant, the trigger `motion.div` and the modal `motion.div` share the same `layoutId` (`modal`), so Framer Motion morphs one into the other. The close icon also uses a shared `layoutId` (`icon`) so the Plus icon seamlessly rotates into the dismiss button. Use this component when the visual connection between a compact trigger and the resulting dialog adds meaning:

- **"Add item" actions.** A small circular plus button morphs into a "New Project" form. The morph communicates that the form is the expansion of that button, not a separate page.
- **Image lightboxes.** A thumbnail trigger expands into a full-size preview dialog. Pair with `variant="scale"` to preserve the spatial relationship between the thumbnail and the overlay.
- **Confirmation dialogs.** A delete icon expands into a confirmation panel. `variant="fade-in"` (tween 0.3s easeOut) is calmer than a bouncy spring for destructive actions.
- **Quick forms.** Contact, feedback, or subscribe forms that should feel embedded in the page rather than navigated to. `variant="slide-bottom"` (initial `y: "100%"`) suits mobile-first layouts where bottom sheets are familiar.
- **Side panel details.** Clicking a table row triggers a `variant="slide-right"` panel (initial `x: "100%"`) that reveals the row's full record without a page change.

## When to use / when not to use

**Use `SharedModal` when:**

- There is one trigger and one corresponding modal, and the visual morph between them is meaningful to the user.
- The modal content is relatively simple: a form with two or three fields, a confirmation message, or a detail preview. The component renders a fixed-height panel with a scrollable body; deeply nested UIs are awkward here.
- The trigger is small and centered on screen, so the `"scale"` morph has room to expand without jumping off-screen.

**Do not use `SharedModal` when:**

- You need to open the same modal from multiple, differently-shaped triggers. Framer Motion's `layoutId` ties the morph to a single trigger element; shared IDs across multiple triggers cause the animation to jump between them.
- The dialog needs to be fully accessible immediately (a screen reader or keyboard-only workflow). The current implementation has no focus trap, no `role="dialog"`, and no Escape key handler. Use a headless dialog primitive (Radix UI `Dialog`, Headless UI `Dialog`) with a custom `AnimatePresence` wrapper instead.
- You want a slide-over drawer that persists across navigations. A drawer component with `position: fixed` and controlled open state is a better fit.
- The morph effect is not worth the complexity. For a plain centered modal without a trigger-to-dialog animation, the overhead of shared `layoutId` is unnecessary. A simple `AnimatePresence` fade is cleaner.

Compare with [Expanding Card](/docs/transitions/expanding-card): `ExpandingCard` morphs one item out of a grid into a detail panel; `SharedModal` morphs a single floating trigger into a dialog. If you have a grid of triggers, use `ExpandingCard`.

## Accessibility notes

The current implementation needs these additions before it is safe for keyboard and screen-reader users:

- **Role and label.** Add `role="dialog"` and `aria-modal="true"` to the expanded `motion.div`. Add `aria-labelledby` pointing to the modal's `<h3>` id so screen readers announce the dialog title when it opens.
- **Focus trap.** On open, move focus to the first interactive element inside the modal (the Cancel button or the close button). Use `focus-trap-react` or manually call `.focus()` inside the `animate` callback. Without this, keyboard users remain focused on the backdrop.
- **Escape to close.** Add a `keydown` listener on `document` for `Escape` that calls `setIsOpen(false)`. Remove the listener on unmount.
- **Return focus.** Store a ref to the trigger element before opening and restore focus to it when the dialog closes. This is especially important when the trigger is re-rendered after the `AnimatePresence` exit because the trigger `motion.div` only exists when `!isOpen`.
- **Scroll lock.** While the modal is open, set `document.body.style.overflow = "hidden"` to prevent background scroll. Restore it on close.
- **Close button label.** The rotated Plus icon used as a close button has no accessible text. Add `aria-label="Close dialog"` to that `<button>`.

## Performance considerations

- **Single `layoutId` constraint.** The `"scale"` variant connects the trigger and modal via a shared `layoutId`. Framer Motion measures both elements before and after the state change to compute the layout delta. This is a single measurement, but it takes a synchronous read of the DOM. Keep the trigger and modal subtrees shallow.
- **`AnimatePresence` mount and unmount.** The trigger and the modal are mutually exclusive inside `AnimatePresence`. When opening, the trigger exits (unmounts) and the modal enters (mounts). On close, the reverse happens. This means the modal's children re-render from scratch on every open. Avoid putting expensive computations directly in the modal's render; memoize or fetch data before mounting.
- **`backdrop-blur-sm` backdrop.** The `bg-black/20 backdrop-blur-sm` overlay is GPU-composited. On mobile browsers and devices with many stacking contexts, blur compositing is expensive. Test on real hardware and remove the blur if needed.
- **`transformPerspective: 1000` style.** This is set directly on the modal `motion.div`. It creates a stacking context and can prevent certain CSS optimizations. Only relevant if you are layering many 3D-transformed elements.
- **Variants `"slide-bottom"` and `"slide-right"`.** These do not use `layoutId`; they use plain `initial`/`animate`/`exit` values (`y: "100%"` or `x: "100%"`). This is cheaper than shared layout because no measurement is needed, just a transform animation.

## Real-world examples

### Image lightbox

A thumbnail expands into a full-size overlay. The `"scale"` variant keeps the image's position in space while it grows.

```tsx
"use client"
import { useState, useId } from "react"
import { motion, AnimatePresence } from "motion/react"

export function ImageLightbox({ src, alt }: { src: string; alt: string }) {
  const [open, setOpen] = useState(false)
  const id = useId()

  return (
    <>
      <AnimatePresence>
        {!open && (
          <motion.img
            layoutId={`${id}-img`}
            src={src}
            alt={alt}
            onClick={() => setOpen(true)}
            className="w-48 h-32 object-cover rounded-lg cursor-pointer shadow-md"
          />
        )}
      </AnimatePresence>
      <AnimatePresence>
        {open && (
          <>
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
              onClick={() => setOpen(false)}
              className="fixed inset-0 z-40 bg-black/70"
            />
            <motion.img
              layoutId={`${id}-img`}
              src={src}
              alt={alt}
              className="fixed inset-0 m-auto z-50 max-w-3xl max-h-[80vh] object-contain rounded-xl shadow-2xl"
              transition={{ type: "spring", bounce: 0.15, duration: 0.5 }}
            />
          </>
        )}
      </AnimatePresence>
    </>
  )
}
```
