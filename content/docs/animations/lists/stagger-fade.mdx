## Overview

The `StaggerFadeList` component creates a sequential fade-in effect for list items. Perfect for subtle, elegant content reveals.

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/animations/lists/stagger-fade
    ```
  </Tab>
  <Tab value="Manual">
    ```tsx title="components/animations/lists/stagger-fade.tsx"
    "use client"

    import * as React from "react"
    import { motion, HTMLMotionProps } from "framer-motion"
    import { cn } from "@/lib/utils"

    export interface StaggerFadeListProps extends Omit<HTMLMotionProps<"div">, "children"> {
      children: React.ReactNode[]
      staggerDelay?: number
      duration?: number
      from?: number
      as?: keyof JSX.IntrinsicElements
    }

    export function StaggerFadeList({
      children,
      staggerDelay = 0.1,
      duration = 0.4,
      from = 0,
      as: Component = "div",
      className,
      ...props
    }: StaggerFadeListProps) {
      const childrenArray = React.Children.toArray(children)

      const MotionComponent = motion[Component as keyof typeof motion] as any

      return (
        <MotionComponent className={cn(className)} {...props}>
          {childrenArray.map((child, index) => (
            <motion.div
              key={index}
              initial={{ opacity: from }}
              animate={{ opacity: 1 }}
              transition={{
                duration,
                delay: index * staggerDelay,
                ease: "easeOut",
              }}
            >
              {child}
            </motion.div>
          ))}
        </MotionComponent>
      )
    }
    ```
  </Tab>
</Tabs>

## Usage

```tsx
import { StaggerFadeList } from "@/components/animations/lists/stagger-fade"

const items = ["Item 1", "Item 2", "Item 3"]

export default function Example() {
  return (
    <StaggerFadeList staggerDelay={0.1}>
      {items.map((item, index) => (
        <div key={index} className="p-4 border rounded-lg">
          {item}
        </div>
      ))}
    </StaggerFadeList>
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode[]` | - | Array of items to animate |
| `staggerDelay` | `number` | `0.1` | Delay between each item in seconds |
| `duration` | `number` | `0.4` | Animation duration in seconds |
| `from` | `number` | `0` | Initial opacity value |
| `as` | `keyof JSX.IntrinsicElements` | `"div"` | Container element type |
| `className` | `string` | - | Additional CSS classes |
