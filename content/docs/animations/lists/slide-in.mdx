## Overview

The `SlideInList` component creates a staggered slide-in effect for list items. Perfect for revealing content progressively from any direction.

## Installation

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/lists-slide-in.json
    ```
  </Tab>
  <Tab value="Manual">
    ```tsx title="components/animations/lists/slide-in.tsx"
    "use client"    import * as React from "react"    import { motion, HTMLMotionProps } from "framer-motion"    import { cn } from "@/lib/utils"    export interface SlideInListProps extends Omit<HTMLMotionProps<"div">, "children"> {      children: React.ReactNode[]      direction?: "left" | "right" | "top" | "bottom"      staggerDelay?: number      duration?: number      distance?: number      as?: keyof JSX.IntrinsicElements    }    export function SlideInList({      children,      direction = "left",      staggerDelay = 0.1,      duration = 0.3,      distance = 20,      as: Component = "div",      className,      ...props    }: SlideInListProps) {      const childrenArray = React.Children.toArray(children)      const getInitialPosition = () => {        switch (direction) {          case "left":            return { x: -distance, y: 0 }          case "right":            return { x: distance, y: 0 }          case "top":            return { x: 0, y: -distance }          case "bottom":            return { x: 0, y: distance }        }      }      const MotionComponent = motion[Component as keyof typeof motion] as any      return (        <MotionComponent className={cn(className)} {...props}>          {childrenArray.map((child, index) => (            <motion.div              key={index}              initial={{ ...getInitialPosition(), opacity: 0 }}              animate={{ x: 0, y: 0, opacity: 1 }}              transition={{                duration,                delay: index * staggerDelay,                ease: "easeOut",              }}            >              {child}            </motion.div>          ))}        </MotionComponent>      )    }    ```
  </Tab>
</Tabs>

## Usage

```tsx
import { SlideInList } from "@/components/animations/lists/slide-in"const items = ["Item 1", "Item 2", "Item 3"]export default function Example() {  return (    <SlideInList direction="left" staggerDelay={0.1}>      {items.map((item, index) => (        <div key={index} className="p-4 border rounded-lg">          {item}        </div>      ))}    </SlideInList>  )}```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode[]` | - | Array of items to animate |
| `direction` | `"left"` \| `"right"` \| `"top"` \| `"bottom"` | `"left"` | Direction to slide from |
| `staggerDelay` | `number` | `0.1` | Delay between each item in seconds |
| `duration` | `number` | `0.3` | Animation duration in seconds |
| `distance` | `number` | `20` | Distance to slide in pixels |
| `as` | `keyof JSX.IntrinsicElements` | `"div"` | Container element type |
| `className` | `string` | - | Additional CSS classes |
