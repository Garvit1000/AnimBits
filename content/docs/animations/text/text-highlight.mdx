    <Steps>
      <Step>
        **Install dependencies**
        ```bash
        npm install motion
        ```
      </Step>
      <Step>
        **Copy the component code**
        ```tsx title="components/animations/text/text-highlight.tsx"
        "use client";

        import { cn } from "@/lib/utils";
        import { motion } from "motion/react";

        interface TextHighlightProps {
            children: React.ReactNode;
            className?: string;
            delay?: number;
            duration?: number;
            color?: string;
        }

        export function TextHighlight({
            children,
            className,
            delay = 0,
            duration = 0.6,
            color = "#fde047",
        }: TextHighlightProps) {
            return (
                <span className={cn("relative inline-block pb-1", className)}>
                    {/* Simple marker underline */}
                    <motion.span
                        className="absolute bottom-0 left-0 right-0 h-[8px] rounded-sm"
                        initial={{ scaleX: 0 }}
                        animate={{ scaleX: 1 }}
                        transition={{
                            duration,
                            delay,
                            ease: [0.16, 1, 0.3, 1],
                        }}
                        style={{
                            backgroundColor: color,
                            transformOrigin: "left",
                        }}
                    />

                    {/* Text */}
                    <span className="relative">
                        {children}
                    </span>
                </span>
            );
        }
        ```
      </Step>
    </Steps>
  </Tab>
</Tabs>

## Props

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `children` | `React.ReactNode` | - | The text content to highlight. |
| `className` | `string` | - | Additional CSS classes. |
| `delay` | `number` | `0` | Delay before animation starts (in seconds). |
| `duration` | `number` | `0.6` | Duration of the animation (in seconds). |
| `color` | `string` | `"#fde047"` | The color of the marker underline. |

## Features

- **Simple marker underline**: Clean 8px underline that animates from left to right
- **Smooth animation**: Custom easing for professional motion
- **Fully customizable**: Adjust color, duration, and delay
- **Minimal design**: No glow, shimmer, or background effects - just clean highlighting
