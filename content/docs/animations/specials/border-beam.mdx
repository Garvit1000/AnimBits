## Installation

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/specials-border-beam.json
    ```
  </Tab>
  <Tab value="Manual">
    <Steps>
      <Step>
        **Install dependencies**
        ```bash
        npm install motion
        ```
      </Step>
      <Step>
        **Copy the component code**
        ```tsx title="components/animations/specials/border-beam.tsx"
"use client";

import { cn } from "@/lib/utils";
import { motion } from "motion/react";

interface BorderBeamProps {
  className?: string;
  size?: number;
  duration?: number;
  borderWidth?: number;
  anchor?: number;
  colorFrom?: string;
  colorTo?: string;
  delay?: number;
  radius?: number;
}

export function BorderBeam({
  className,
  size = 200,
  duration = 15,
  anchor = 90,
  borderWidth = 1.5,
  colorFrom = "#ffaa40",
  colorTo = "#9c40ff",
  delay = 0,
  radius = 0,
}: BorderBeamProps) {
  return (
    <div
      style={
        {
          "--size": size,
          "--duration": duration,
          "--anchor": anchor,
          "--border-width": borderWidth,
          "--color-from": colorFrom,
          "--color-to": colorTo,
          "--delay": delay,
          "--radius": radius,
        } as React.CSSProperties
      }
      className={cn(
        "pointer-events-none absolute inset-0 rounded-[inherit] [border:calc(var(--border-width)*1px)_solid_transparent]",
        // mask styles
        "![mask-clip:padding-box,border-box] ![mask-composite:intersect] [mask:linear-gradient(transparent,transparent),linear-gradient(white,white)]",
        // pseudo styles
        "after:absolute after:aspect-square after:w-[calc(var(--size)*1px)] after:animate-border-beam after:[animation-delay:calc(var(--delay)*1s)] after:[background:linear-gradient(to_left,var(--color-from),var(--color-to),transparent)] after:[offset-anchor:calc(var(--anchor)*1%)_50%] after:[offset-path:rect(0_auto_auto_0_round_calc(var(--radius)*1px))]",
        className,
      )}
    />
  );
}
        ```
      </Step>
      <Step>
        **Update tailwind.css**
        Add the following keyframes to your `app/globals.css` or `tailwind.config.ts`:
        ```css
        @keyframes border-beam {
          100% {
            offset-distance: 100%;
          }
        }
        ```
      </Step>
    </Steps>
  </Tab>
</Tabs>

## Props

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `size` | `number` | `200` | The length of the beam. |
| `duration` | `number` | `15` | The duration of the animation in seconds. |
| `borderWidth` | `number` | `1.5` | The width of the border. |
| `colorFrom` | `string` | `"#ffaa40"` | The starting color of the gradient. |
| `colorTo` | `string` | `"#9c40ff"` | The ending color of the gradient. |
| `delay` | `number` | `0` | The delay before the animation starts. |
| `radius` | `number` | `0` | The border radius of the path. Set this to match the container's border radius. |
| `className` | `string` | - | Additional CSS classes. |
