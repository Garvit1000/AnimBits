## Overview

The `LiftButton` component creates a button that lifts up on hover with an optional shadow effect. Built using the `useHoverLift` primitive hook, providing a polished, interactive feel.

<Callout type="info">
  **Uses primitive hook:** This component uses `useHoverLift` internally. You can use that hook directly on any element for the same effect.
</Callout>

  <Tab value="Manual">
    **Step 1: Copy the hook**
    
    ```tsx title="lib/hooks/use-hover-lift.ts"
    import { type MotionProps, type Easing } from "framer-motion";

    export interface UseHoverLiftOptions {
      liftDistance?: number;
      addShadow?: boolean;
      shadowIntensity?: number;
      duration?: number;
      ease?: Easing | Easing[];
      scale?: number;
    }

    export function useHoverLift(options: UseHoverLiftOptions = {}): MotionProps {
      const {
        liftDistance = 8,
        addShadow = true,
        shadowIntensity = 0.3,
        duration = 0.2,
        ease = "easeOut",
        scale = 1,
      } = options;

      const shadowBlur = liftDistance * 2;
      const shadowSpread = Math.round(liftDistance / 4);

      return {
        initial: {
          y: 0,
          scale: 1,
          boxShadow: addShadow
            ? `0 2px 4px rgba(0, 0, 0, ${shadowIntensity * 0.3})`
            : undefined,
        },
        whileHover: {
          y: -liftDistance,
          scale,
          boxShadow: addShadow
            ? `0 ${shadowBlur}px ${shadowBlur + shadowSpread}px rgba(0, 0, 0, ${shadowIntensity})`
            : undefined,
        },
        transition: {
          duration,
          ease,
        },
      };
    }
    ```

    **Step 2: Copy the component**

    ```tsx title="components/animations/buttons/lift.tsx"
    "use client"

    import { motion, type HTMLMotionProps } from "framer-motion"
    import { useHoverLift } from "@/lib/hooks"

    interface LiftButtonProps extends Omit<HTMLMotionProps<"button">, "animate"> {
      children: React.ReactNode
      liftHeight?: number
      shadow?: boolean
    }

    export function LiftButton({
      children,
      liftHeight = 8,
      shadow = true,
      ...props
    }: LiftButtonProps) {
      const liftProps = useHoverLift({
        liftDistance: liftHeight,
        addShadow: shadow,
      })

      return (
        <motion.button {...liftProps} {...props}>
          {children}
        </motion.button>
      )
    }
    ```

## Using the Hook Directly

For maximum flexibility, use the `useHoverLift` hook on any element:

```tsx
import { motion } from "framer-motion"
import { useHoverLift } from "@/lib/hooks"

export function CustomElement() {
  const liftProps = useHoverLift({ liftDistance: 12 })
  
  return (
    <motion.div 
      {...liftProps} 
      className="inline-flex items-center gap-2 rounded-lg bg-white border border-neutral-200 p-6 shadow-sm"
    >
      <h3 className="font-semibold">Any Element</h3>
      <p className="text-sm text-neutral-600">Can have lift effect</p>
    </motion.div>
  )
}
```

## Combining with Other Effects

```tsx
import { motion } from "framer-motion"
import { useHoverLift, useHoverGlow } from "@/lib/hooks"

export function LiftAndGlow() {
  const lift = useHoverLift({ liftDistance: 10 })
  const glow = useHoverGlow({ glowColor: "rgba(59, 130, 246, 0.5)" })
  
  const combined = {
    ...lift,
    whileHover: {
      ...lift.whileHover,
      ...glow.whileHover,
    },
  }
  
  return (
    <motion.button {...combined} className="inline-flex h-11 items-center gap-2 rounded-lg bg-blue-600 px-6 text-sm font-semibold text-white">
      Lift + Glow
    </motion.button>
  )
}
```
