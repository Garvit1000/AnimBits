    **Step 2: Copy the component**

    ```tsx title="components/animations/buttons/fade-in.tsx"
    "use client"

    import { motion, type HTMLMotionProps } from "framer-motion"
    import { fadeInVariants, createDelayedTransition } from "@/lib/animation-presets"

    interface FadeInButtonProps extends Omit<HTMLMotionProps<"button">, "animate"> {
      children: React.ReactNode
      delay?: number
      duration?: number
    }

    export function FadeInButton({
      children,
      delay = 0,
      duration = 0.3,
      ...props
    }: FadeInButtonProps) {
      return (
        <motion.button
          variants={fadeInVariants}
          initial="hidden"
          animate="visible"
          transition={createDelayedTransition(delay, duration)}
          {...props}
        >
          {children}
        </motion.button>
      )
    }
    ```

## Usage

```tsx
import { FadeInButton } from "@/components/animations/buttons/fade-in"

<FadeInButton className="inline-flex h-11 items-center gap-2 rounded-lg bg-blue-600 px-6 text-sm font-semibold text-white">
  Hover Me
</FadeInButton>
```

### With Delay

```tsx
<FadeInButton delay={0.5} className="inline-flex h-11 items-center gap-2 rounded-lg bg-blue-600 px-6 text-sm font-semibold text-white">
  Delayed Fade
</FadeInButton>
```

### Staggered Buttons

```tsx
<div className="flex gap-4">
  <FadeInButton delay={0}>First</FadeInButton>
  <FadeInButton delay={0.2}>Second</FadeInButton>
  <FadeInButton delay={0.4}>Third</FadeInButton>
</div>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `delay` | `number` | `0` | Delay before animation starts (seconds) |
| `duration` | `number` | `0.3` | Animation duration (seconds) |
| `children` | `React.ReactNode` | Required | Button content |
