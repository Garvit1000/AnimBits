## Overview

Card components with various animation effects.

## Available Cards

### Hover Lift Card

```tsx
import { CardHoverLift } from "@/components/animations/cards/hover-lift"

<CardHoverLift>
  <h3>Card Title</h3>
  <p>Card content</p>
</CardHoverLift>
```

### Hover Glow Card

```tsx
import { CardHoverGlow } from "@/components/animations/cards/hover-glow"

<CardHoverGlow>
  <h3>Glowing Card</h3>
</CardHoverGlow>
```

### Fade In Card

```tsx
import { CardFadeIn } from "@/components/animations/cards/fade-in"

<CardFadeIn>
  <h3>Fades in on mount</h3>
</CardFadeIn>
```
