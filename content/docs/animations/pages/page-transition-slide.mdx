## Overview

The `PageTransitionSlide` component creates a smooth sliding effect when navigating between pages. Choose from left, right, top, or bottom directions. Uses the native View Transitions API for hardware-accelerated animations with customizable duration and easing.

    **Step 1:** Create the component

    ```tsx title="components/animations/pages/page-transition-slide.tsx"
    "use client"

    import * as React from "react"
    import { usePathname } from "next/navigation"

    export type SlideDirection = "left" | "right" | "top" | "bottom"

    export interface PageTransitionSlideProps {
      children: React.ReactNode
      direction?: SlideDirection
      duration?: number
      ease?: string
    }

    export function PageTransitionSlide({
      children,
      direction = "left",
      duration = 0.4,
      ease = "ease-in-out",
    }: PageTransitionSlideProps) {
      const pathname = usePathname()

      React.useEffect(() => {
        if (!document.startViewTransition) return

        document.documentElement.style.setProperty('--page-transition-duration', `${duration}s`)
        document.documentElement.style.setProperty('--page-transition-ease', ease)
        document.documentElement.setAttribute('data-slide-direction', direction)
      }, [pathname, duration, ease, direction])

      return <>{children}</>
    }
    ```

    **Step 2:** Create navigation wrapper

    ```tsx title="components/animations/pages/page-transition-link.tsx"
    "use client"

    import * as React from "react"
    import { useRouter } from "next/navigation"
    import Link, { LinkProps } from "next/link"

    interface PageTransitionLinkProps extends LinkProps {
      children: React.ReactNode
      className?: string
    }

    export function PageTransitionLink({
      href,
      children,
      className,
      ...props
    }: PageTransitionLinkProps) {
      const router = useRouter()

      const handleClick = async (e: React.MouseEvent<HTMLAnchorElement>) => {
        e.preventDefault()

        if (!document.startViewTransition) {
          router.push(href.toString())
          return
        }

        const transition = document.startViewTransition(() => {
          router.push(href.toString())
        })

        await transition.ready
      }

      return (
        <Link 
          href={href} 
          onClick={handleClick} 
          className={className}
          {...props}
        >
          {children}
        </Link>
      )
    }
    ```

## Usage

### Setup in Layout

Wrap your application with the `PageTransitionSlide` component:

```tsx title="app/layout.tsx"
import { PageTransitionSlide } from "@/components/animations/pages/page-transition-slide"

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="en">
      <body>
        <PageTransitionSlide direction="left">
          {children}
        </PageTransitionSlide>
      </body>
    </html>
  )
}
```

### Using Transition Links

Replace standard Next.js `Link` components with `PageTransitionLink`:

```tsx
import { PageTransitionLink } from "@/components/animations/pages/page-transition-link"

export function Navigation() {
  return (
    <nav className="flex gap-4">
      <PageTransitionLink href="/" className="hover:text-blue-600">
        Home
      </PageTransitionLink>
      <PageTransitionLink href="/about" className="hover:text-blue-600">
        About
      </PageTransitionLink>
      <PageTransitionLink href="/contact" className="hover:text-blue-600">
        Contact
      </PageTransitionLink>
    </nav>
  )
}
```

### Programmatic Navigation

```tsx
"use client"

import { useRouter } from "next/navigation"

export function NavigateButton() {
  const router = useRouter()

  const handleNavigate = async () => {
    if (!document.startViewTransition) {
      router.push('/destination')
      return
    }

    const transition = document.startViewTransition(() => {
      router.push('/destination')
    })

    await transition.ready
  }

  return (
    <button onClick={handleNavigate} className="inline-flex h-11 items-center gap-2 rounded-lg bg-blue-600 px-6 text-sm font-semibold text-white">
      Navigate
    </button>
  )
}
```

## Props

### PageTransitionSlide

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | The page content |
| `direction` | `"left"` \| `"right"` \| `"top"` \| `"bottom"` | `"left"` | Slide direction |
| `duration` | `number` | `0.4` | Transition duration in seconds |
| `ease` | `string` | `"ease-in-out"` | CSS easing function |

### PageTransitionLink

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `href` | `string` | - | Destination URL |
| `children` | `ReactNode` | - | Link content |
| `className` | `string` | - | Additional CSS classes |
| `...props` | `LinkProps` | - | All Next.js Link props |

## Advanced Usage

### Dynamic Direction Based on Navigation

You can change the slide direction based on the navigation path:

```tsx
"use client"

import { useState, useEffect } from "react"
import { usePathname } from "next/navigation"
import { PageTransitionSlide, SlideDirection } from "@/components/animations/pages/page-transition-slide"

export function DynamicSlideTransition({ children }: { children: React.ReactNode }) {
  const pathname = usePathname()
  const [direction, setDirection] = useState<SlideDirection>("left")

  useEffect(() => {
    // Example: slide right when going back
    if (pathname.includes('/back')) {
      setDirection('right')
    } else {
      setDirection('left')
    }
  }, [pathname])

  return (
    <PageTransitionSlide direction={direction}>
      {children}
    </PageTransitionSlide>
  )
}
```

### Parallax Effect

Create a parallax effect by having different slide speeds:

```css
::view-transition-old(root) {
  animation: slide-out-left 0.4s ease-in-out;
}

::view-transition-new(root) {
  animation: slide-in-left 0.6s ease-in-out;
}
```

### Slide with Fade

Combine slide and fade for a more dynamic effect:

```css
[data-slide-direction="left"]::view-transition-old(root) {
  animation: 
    slide-out-left var(--page-transition-duration) var(--page-transition-ease),
    fade-out var(--page-transition-duration) var(--page-transition-ease);
}

[data-slide-direction="left"]::view-transition-new(root) {
  animation: 
    slide-in-left var(--page-transition-duration) var(--page-transition-ease),
    fade-in var(--page-transition-duration) var(--page-transition-ease);
}

@keyframes fade-out {
  to { opacity: 0; }
}

@keyframes fade-in {
  from { opacity: 0; }
}
```

## Navigation History Integration

Track navigation direction based on browser history:

```tsx
"use client"

import { useState, useEffect } from "react"
import { usePathname } from "next/navigation"
import { PageTransitionSlide, SlideDirection } from "@/components/animations/pages/page-transition-slide"

export function HistoryAwareSlide({ children }: { children: React.ReactNode }) {
  const pathname = usePathname()
  const [direction, setDirection] = useState<SlideDirection>("left")
  const [prevPath, setPrevPath] = useState(pathname)

  useEffect(() => {
    // Simple heuristic: if path is shorter, likely going back
    if (pathname.length < prevPath.length) {
      setDirection('right')
    } else {
      setDirection('left')
    }
    setPrevPath(pathname)
  }, [pathname, prevPath])

  return (
    <PageTransitionSlide direction={direction}>
      {children}
    </PageTransitionSlide>
  )
}
```

## Performance Tips

1. **Use transforms**: Transforms are GPU-accelerated and more performant than position changes
2. **Keep transitions quick**: 300-500ms is ideal for slide transitions
3. **Avoid large pages**: Very tall/wide pages may cause jank during slides
4. **Test on mobile**: Ensure smooth performance on lower-end devices
5. **Use `will-change`**: Add `will-change: transform` for better performance

```css
::view-transition-old(root),
::view-transition-new(root) {
  will-change: transform;
}
```

## Troubleshooting

### Transition doesn't work

1. Check browser support (Chrome 111+, Edge 111+, Opera 97+)
2. Verify CSS is properly imported in `globals.css`
3. Ensure `document.startViewTransition` is available
4. Check console for JavaScript errors

### Janky animation

1. Reduce transition duration
2. Use simpler easing functions
3. Check for heavy JavaScript on page load
4. Reduce page complexity during transition

### Wrong direction

1. Verify `data-slide-direction` attribute is set correctly
2. Check that CSS matches the direction attribute
3. Clear browser cache and reload
