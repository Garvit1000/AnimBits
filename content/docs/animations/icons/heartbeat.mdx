## Overview

The `HeartbeatIcon` component creates a rhythmic heartbeat animation with configurable BPM (beats per minute). Perfect for health apps, activity indicators, and live status displays.

    **Step 2: Copy the component**

    ```tsx title="components/animations/icons/heartbeat.tsx"
    "use client"

    import * as motion from "framer-motion/client"
    import type { HTMLMotionProps } from "framer-motion"
    import { heartbeatVariants } from "@/lib/animation-presets"

    interface HeartbeatIconProps extends Omit<HTMLMotionProps<"div">, "animate"> {
      children: React.ReactNode
      bpm?: number
    }

    export function HeartbeatIcon({
      children,
      bpm = 60,
      ...props
    }: HeartbeatIconProps) {
      // Convert BPM to duration (one complete heartbeat cycle)
      const duration = 60 / bpm

      return (
        <motion.div
          variants={heartbeatVariants}
          initial="initial"
          animate="animate"
          transition={{
            duration,
            ease: "easeInOut",
            repeat: Infinity,
          }}
          {...props}
        >
          {children}
        </motion.div>
      )
    }
    ```

## Usage

```tsx
import { HeartbeatIcon } from "@/components/animations/icons/heartbeat"
import { Heart } from "lucide-react"

export default function Example() {
  return (
    <HeartbeatIcon bpm={60}>
      <Heart className="w-6 h-6 text-red-500 fill-red-500" />
    </HeartbeatIcon>
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | The icon element to animate |
| `bpm` | `number` | `60` | Beats per minute (controls animation speed) |
