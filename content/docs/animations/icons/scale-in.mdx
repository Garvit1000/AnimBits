## Overview

The `ScaleInIcon` component creates a scale-in entrance animation. Perfect for revealing icons with a smooth zoom effect.

    **Step 2: Copy the component**

    ```tsx title="components/animations/icons/scale-in.tsx"
"use client";
import * as motion from "framer-motion/client";
import type { HTMLMotionProps } from "motion/react";
import {
  scaleInVariants,
  createDelayedTransition,
} from "@/lib/animation-presets";
interface ScaleInIconProps extends Omit<HTMLMotionProps<"div">, "animate"> {
  children: React.ReactNode;
  delay?: number;
  duration?: number;
  from?: number;
}
export function ScaleInIcon({
  children,
  delay = 0,
  duration = 0.3,
  from = 0.9,
  ...props
}: ScaleInIconProps) {
  const customVariants = {
    hidden: { opacity: 0, scale: from },
    visible: { opacity: 1, scale: 1 },
  };
  return (
    <motion.div
      variants={customVariants}
      initial="hidden"
      animate="visible"
      transition={createDelayedTransition(delay, duration)}
      {...props}
    >
      {children}
    </motion.div>
  );
}
```

## Usage

```tsx
import { ScaleInIcon } from "@/components/animations/icons/scale-in"
import { Zap } from "lucide-react"

export default function Example() {
  return (
    <ScaleInIcon delay={0.1} duration={0.3} from={0.8}>
      <Zap className="w-6 h-6" />
    </ScaleInIcon>
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | The icon element to animate |
| `delay` | `number` | `0` | Delay before animation starts in seconds |
| `duration` | `number` | `0.3` | Animation duration in seconds |
| `from` | `number` | `0.9` | Initial scale value |
