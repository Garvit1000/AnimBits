## Overview

The `MorphIcon` component smoothly morphs between different SVG path states. Perfect for play/pause toggles, menu/close buttons, and state transitions.

## Installation

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/icons-morph.json
    ```
  </Tab>
  <Tab value="Manual">
    **Step 1: Add the utility**

    ```tsx title="lib/utils.ts"
    import { clsx, type ClassValue } from "clsx"
    import { twMerge } from "tailwind-merge"

    export function cn(...inputs: ClassValue[]) {
      return twMerge(clsx(inputs))
    }
    ```

    **Step 2: Copy the component**

    ```tsx title="components/animations/icons/morph.tsx"
    "use client"

    import * as React from "react"
    import * as motion from "framer-motion/client"
    import type { SVGMotionProps } from "framer-motion"
    import { cn } from "@/lib/utils"

    export interface MorphIconProps extends Omit<SVGMotionProps<SVGSVGElement>, "children"> {
      paths: string[]
      duration?: number
      interval?: number
      strokeColor?: string
      strokeWidth?: number
    }

    export function MorphIcon({
      paths,
      className,
      duration = 0.3,
      interval = 2,
      strokeColor = "currentColor",
      strokeWidth = 2,
      ...props
    }: MorphIconProps) {
      const [currentIndex, setCurrentIndex] = React.useState(0)

      React.useEffect(() => {
        const timer = setInterval(() => {
          setCurrentIndex((prev) => (prev + 1) % paths.length)
        }, interval * 1000)

        return () => clearInterval(timer)
      }, [paths.length, interval])

      return (
        <motion.svg
          className={cn(className)}
          fill="none"
          stroke={strokeColor}
          strokeWidth={strokeWidth}
          strokeLinecap="round"
          strokeLinejoin="round"
          {...props}
        >
          <motion.path
            d={paths[currentIndex]}
            animate={{ d: paths[currentIndex] }}
            transition={{ duration, ease: "easeInOut" }}
          />
        </motion.svg>
      )
    }
    ```
  </Tab>
</Tabs>

## Usage

```tsx
import { MorphIcon } from "@/components/animations/icons/morph"

export default function Example() {
  return (
    <MorphIcon 
      paths={[
        "M5 3l14 9-14 9V3z", // play
        "M6 4h4v16H6V4zm8 0h4v16h-4V4z" // pause
      ]}
      viewBox="0 0 24 24"
      className="w-6 h-6"
    />
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `paths` | `string[]` | - | Array of SVG path data strings to morph between |
| `duration` | `number` | `0.3` | Animation duration in seconds |
| `interval` | `number` | `2` | Interval between morphs in seconds |
| `strokeColor` | `string` | `"currentColor"` | Stroke color |
| `strokeWidth` | `number` | `2` | Stroke width |
| `viewBox` | `string` | - | SVG viewBox attribute |
| `className` | `string` | - | Additional CSS classes |
