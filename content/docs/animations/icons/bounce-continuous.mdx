## Overview

The `BounceContinuous` component creates a continuous bouncing animation. Perfect for scroll indicators, download icons, and attention-grabbing elements.

    **Step 2: Copy the component**

    ```tsx title="components/animations/icons/bounce-continuous.tsx"
    "use client"    import * as motion from "framer-motion/client"    import type { HTMLMotionProps } from "framer-motion"    import { bounceContinuousVariants, createContinuousTransition } from "@/lib/animation-presets"    interface BounceContinuousProps extends Omit<HTMLMotionProps<"div">, "animate"> {      children: React.ReactNode      duration?: number    }    export function BounceContinuous({      children,      duration = 1,      ...props    }: BounceContinuousProps) {      return (        <motion.div          variants={bounceContinuousVariants}          initial="initial"          animate="animate"          transition={createContinuousTransition(duration, "easeInOut")}          {...props}        >          {children}        </motion.div>      )    }    ```

## Usage

```tsx
import { BounceContinuous } from "@/components/animations/icons/bounce-continuous"import { ArrowDown } from "lucide-react"export default function Example() {  return (    <BounceContinuous duration={1}>      <ArrowDown className="w-6 h-6" />    </BounceContinuous>  )}```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | The icon element to animate |
| `duration` | `number` | `1` | Duration of one bounce cycle in seconds |
