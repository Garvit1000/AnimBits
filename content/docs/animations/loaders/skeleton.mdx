## Overview

The `LoaderSkeleton` component creates a skeleton placeholder with a shimmer effect. Perfect for content loading states and placeholder UI.

## Installation

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/loaders-skeleton.json
    ```
  </Tab>
  <Tab value="Manual">
    ```tsx title="components/animations/loaders/skeleton.tsx"
    "use client"

    import * as React from "react"
    import { motion, HTMLMotionProps } from "framer-motion"
    import { cn } from "@/lib/utils"

    export interface LoaderSkeletonProps extends HTMLMotionProps<"div"> {
      width?: string | number
      height?: number
      borderRadius?: number
      baseColor?: string
      highlightColor?: string
      duration?: number
    }

    export function LoaderSkeleton({
      className,
      width = "100%",
      height = 20,
      borderRadius = 4,
      baseColor = "#e5e7eb",
      highlightColor = "#f3f4f6",
      duration = 1.5,
      ...props
    }: LoaderSkeletonProps) {
      return (
        <div
          className={cn("relative overflow-hidden", className)}
          style={{
            width: width,
            height: height,
            borderRadius: borderRadius,
            backgroundColor: baseColor,
          }}
        >
          <motion.div
            className="absolute inset-0"
            style={{
              background: `linear-gradient(90deg, transparent, ${highlightColor}, transparent)`,
            }}
            animate={{
              x: ["-100%", "100%"],
            }}
            transition={{
              duration,
              ease: "easeInOut",
              repeat: Infinity,
            }}
          />
        </div>
      )
    }
    ```
  </Tab>
</Tabs>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `width` | `string \| number` | `"100%"` | Width of the skeleton |
| `height` | `number` | `20` | Height of the skeleton in pixels |
| `borderRadius` | `number` | `4` | Border radius in pixels |
| `baseColor` | `string` | `"#e5e7eb"` | Base color of the skeleton |
| `highlightColor` | `string` | `"#f3f4f6"` | Highlight color of the skeleton |
| `duration` | `number` | `1.5` | Animation duration in seconds |
