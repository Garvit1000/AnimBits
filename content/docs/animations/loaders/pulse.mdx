## Overview

The `LoaderPulse` component creates a pulsing circle animation. Perfect for subtle loading indicators and status displays.

## Installation

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/loaders-pulse.json
    ```
  </Tab>
  <Tab value="Manual">
    **Step 1: Add the utility**

    ```tsx title="lib/utils.ts"
    import { clsx, type ClassValue } from "clsx"
    import { twMerge } from "tailwind-merge"

    export function cn(...inputs: ClassValue[]) {
      return twMerge(clsx(inputs))
    }
    ```

    **Step 2: Copy the component**

    ```tsx title="components/animations/loaders/pulse.tsx"
"use client";
import * as React from "react";
import { motion, HTMLMotionProps } from "framer-motion";
import { cn } from "@/lib/utils";
export interface LoaderPulseProps extends HTMLMotionProps<"div"> {
  size?: number;
  color?: string;
  duration?: number;
}
export function LoaderPulse({
  className,
  size = 60,
  color = "currentColor",
  duration = 1.5,
  ...props
}: LoaderPulseProps) {
  return (
    <motion.div
      className={cn("rounded-full", className)}
      style={{
        width: size,
        height: size,
        backgroundColor: color,
      }}
      animate={{
        scale: [1, 1.2, 1],
        opacity: [0.5, 1, 0.5],
      }}
      transition={{
        duration,
        ease: "easeInOut",
        repeat: Infinity,
      }}
      {...props}
    />
  );
}

```
  </Tab>
</Tabs>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `number` | `60` | Size of the circle in pixels |
| `color` | `string` | `"currentColor"` | Color of the pulse |
| `duration` | `number` | `1.5` | Animation duration in seconds |
