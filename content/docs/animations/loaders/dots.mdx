## Overview

The `LoaderDots` component creates a bouncing dots animation. Perfect for loading states with a playful feel.

## Installation

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn add https://animbits.dev/r/@animbits/loaders/dots.json
    ```
  </Tab>
  <Tab value="Manual">
    **Step 1: Add the utility**

    ```tsx title="lib/utils.ts"
    import { clsx, type ClassValue } from "clsx"
    import { twMerge } from "tailwind-merge"

    export function cn(...inputs: ClassValue[]) {
      return twMerge(clsx(inputs))
    }
    ```

    **Step 2: Copy the component**

    ```tsx title="components/animations/loaders/dots.tsx"
    "use client"

    import * as React from "react"
    import { motion, HTMLMotionProps } from "framer-motion"
    import { cn } from "@/lib/utils"

    export interface LoaderDotsProps extends Omit<HTMLMotionProps<"div">, "children"> {
      size?: number
      color?: string
      count?: number
      gap?: number
      duration?: number
    }

    export function LoaderDots({
      className,
      size = 10,
      color = "currentColor",
      count = 3,
      gap = 8,
      duration = 1.2,
      ...props
    }: LoaderDotsProps) {
      return (
        <div className={cn("flex items-center", className)} style={{ gap }}>
          {Array.from({ length: count }).map((_, index) => (
            <motion.div
              key={index}
              className="rounded-full"
              style={{
                width: size,
                height: size,
                backgroundColor: color,
              }}
              animate={{
                y: [0, -size * 1.5, 0],
              }}
              transition={{
                duration,
                ease: "easeInOut",
                repeat: Infinity,
                delay: index * (duration / count),
              }}
            />
          ))}
        </div>
      )
    }
    ```
  </Tab>
</Tabs>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `number` | `10` | Size of each dot in pixels |
| `color` | `string` | `"currentColor"` | Color of the dots |
| `count` | `number` | `3` | Number of dots |
| `gap` | `number` | `8` | Gap between dots in pixels |
| `duration` | `number` | `1.2` | Animation duration in seconds |
